# Matches PO number sequences, e.g.:
#   "Open POs 4521626611, 4521643040, and 4521618014 are affected."
#   "PO# 4521626611"  /  "Purchase Order 4521643040"
_SC_PO_PATTERN = re.compile(
    r"(?:Open\s+)?P\.?O\.?s?\s*(?:#\s*)?(?:\d[\d,\s]*(?:and\s+)?\d{8,12}|\d{8,12})"
    r"|(?:P\.?O\.?|Purchase\s+Order)\s*#?\s*\d{8,12}",
    re.IGNORECASE,
)

# Matches inline inventory/demand quantity references, e.g.:
#   "60 units on open order"  /  "on-hand inventory of 0"  /  "demand of 120 units"
#   "zero on-hand inventory"  /  "0 on-hand"
_SC_QTY_PATTERN = re.compile(
    # Numeric quantity + supply/demand context
    r"\b\d+\s+(?:units?|pcs|pieces|qty)\s+(?:on[\-\s]hand|on[\-\s]order|in\s+stock|of\s+inventory|demand\b)"
    r"|on[\-\s]hand\s+(?:inventory\s+)?(?:of\s+)?\d+"
    r"|demand\s+(?:of\s+)?\d+\s*(?:units?|pcs|pieces)?"
    r"|inventory\s+(?:of\s+)?\d+\s*(?:units?|pcs|pieces|qty)?"
    # Text forms: "zero on-hand inventory", "zero inventory", "0 on-hand"

def _select_sc_generic_statement(user_text: str) -> Tuple[str, str]:
    """
    Scans user_text and returns the single best-matching approved SC generic statement.
    Returns (label, full_statement_text). Falls back to 'Short-Term Coverage Only'.
    """
    for label, patterns, statement in _SC_STATEMENT_MAP:
        if not patterns:
            return label, statement  # default entry
        for pattern in patterns:
            if re.search(pattern, user_text, re.IGNORECASE):
                return label, statement
    last = _SC_STATEMENT_MAP[-1]
    return last[0], last[2]

def _select_problem_category_and_title_label(
    user_text: str,
    supply_chain_scenario: bool = False,
) -> Tuple[str, str]:
    """
    Selects one taxonomy category and one approved title label outside the LLM.
    This avoids sending full taxonomy and lookup tables on every request.
    """
    text = (user_text or "").lower()
    has_bom_signal = bool(re.search(r"\bbom\b", text))
    has_supply_signal = bool(
        re.search(r"\b(inventory|demand|shortage|coverage|supply\s+risk|supply\s+chain|alternate\s+supplier|lead\s+time)\b", text)
    )
    has_oem_signal = bool(re.search(r"\boem\b", text))
    # Treat explicit OEM EOL/obsolescence phrases as supply-chain unless BOM context is explicit.
    has_eol_signal = bool(
        re.search(
            r"\b(oem\s+eol\s+declaration|oem\s+eol|eol|end[\s\-]+of[\s\-]+life|obsolete(?:d)?|obsolescence)\b",
            text,
        )
    )

    # Supply-chain branch first because it has explicit title labels.
    # Trigger on: explicit SC scenario flag, supply keywords, OR any EOL/obsolescence signal
    # (OEM keyword is NOT required to enter this branch).
    if supply_chain_scenario or has_supply_signal or (has_eol_signal and not has_bom_signal):
        if has_eol_signal:
            if has_oem_signal:
                # OEM-declared EOL — use OEM-specific titles.
                if re.search(r"\b(replacement\s+not\s+yet\s+identified|no\s+replacement|not\s+identified)\b", text):
                    return (
                        "Supply Chain Issues / EOL - Replacement Not Yet Identified",
                        "OEM EOL Declaration - Replacement P/N Not Yet Identified",
                    )
                if re.search(r"\b(supplier\s+proposed\s+replacement|pending\s+approval|proposed\s+replacement)\b", text):
                    return (
                        "Supply Chain Issues / EOL - Supplier Proposed Replacement",
                        "OEM EOL Declaration - Supplier Proposed Replacement Pending Approval",
                    )
                return (
                    "Supply Chain Issues / EOL - Replacement Not Yet Identified",
                    "OEM EOL Declaration",
                )
            else:
                # EOL / Obsolescence without explicit OEM attribution.
                if re.search(r"\b(replacement\s+not\s+yet\s+identified|no\s+replacement|not\s+identified)\b", text):
                    return (
                        "Supply Chain Issues / EOL - Replacement Not Yet Identified",
                        "EOL - Replacement P/N Not Yet Identified",
                    )
                if re.search(r"\b(supplier\s+proposed\s+replacement|pending\s+approval|proposed\s+replacement)\b", text):
                    return (
                        "Supply Chain Issues / EOL - Supplier Proposed Replacement",
                        "EOL - Supplier Proposed Replacement Pending Approval",
                    )
                return (
                    "Supply Chain Issues / EOL - Replacement Not Yet Identified",
                    "OEM EOL Declaration by Supplier",
                )
        if re.search(r"\b(alternate\s+supplier|supply\s+disruption|supplier\s+disruption)\b", text):
            return (
                "Supply Chain Issues / Supply Chain Disruption - Alternate Supplier",
                "Supply Disruption - Alternate Supplier Required",
            )
        if re.search(r"\b(long\s+lead|lead\s+time)\b", text):
            return (
                "Supply Chain Issues / Long Lead Time Component Constraint",
                "Long Lead Time Constraint",
            )
        return (
            "Supply Chain Issues / Inventory vs Demand Mismatch",
            "Supply Chain Criticality - Inventory & Demand Risk",
        )

    # BOM branch.
    if has_bom_signal:
        if re.search(r"\b(missing\s+child|child\s+missing)\b", text):
            return (
                "BOM / Structure & Accuracy / Missing Child Component",
                "Missing BOM Child Component",
            )
        if re.search(r"\b(parent.?child|parent\s+child\s+relationship|where\s+used)\b", text):
            return (
                "BOM / Structure & Accuracy / Wrong Parent-Child Relationship",
                "Incorrect BOM Parent-Child Relationship",
            )
        if re.search(r"\b(obsolete\s+part\s+used|obsolete\s+p/?n\s+in\s+bom)\b", text):
            return (
                "BOM / Part Information / Obsolete Part Used",
                "Obsolete P/N in BOM",
            )
        return (
            "BOM / Structure & Accuracy / Incorrect BOM Item / Child",
            "Incorrect BOM Child Component",
        )

    # DWG branch.
    if re.search(r"\b(gd&t|gdt|datum|feature\s+control\s+frame)\b", text):
        return (
            "DWG / GD&T Related / GD&T Issues",
            "GD&T Issue on DWG",
        )
    if re.search(r"\b(missing\s+dimension|dimensions?\s+missing)\b", text):
        return (
            "DWG / Dimensions & Tolerances / Missing Dimensions",
            "Missing Dimensions on DWG",
        )
    if re.search(r"\b(missing\s+tolerance|incorrect\s+tolerance|over-tight\s+tolerance|stack-?up\s+tolerance)\b", text):
        return (
            "DWG / Dimensions & Tolerances / Missing or Incorrect Tolerance",
            "Missing Tolerance on DWG",
        )
    if re.search(r"\b(revision\s+mismatch|rev\s+mismatch|revision\s+block)\b", text):
        return (
            "DWG / Revision & Configuration / DWG Revision Mismatch",
            "DWG Revision Mismatch",
        )

    return (_DEFAULT_SELECTED_CATEGORY, _DEFAULT_TITLE_LABEL)

def extract_part_numbers(text: str) -> List[str]:
    if not text:
        return []
    seen = set()
    out: List[str] = []
    for p in PART_NUMBER_PATTERN.findall(text):
        p = p.upper()
        if p not in seen:
            seen.add(p)
            out.append(p)
    return out


def _dedupe_lines(lines: List[str]) -> List[str]:
    seen = set()
    out = []
    for ln in lines:
        key = re.sub(r"\s+", " ", ln.strip().lower())
        if key and key not in seen:
            seen.add(key)
            out.append(ln)  # preserve original indentation
    return out


def extract_bom_relationships(text: str) -> List[Tuple[str, str]]:
    """
    Returns (parent, child). Rule: Used in Part Number = Parent.
    """
    if not text:
        return []

    rels: List[Tuple[str, str]] = []
    seen: set[Tuple[str, str]] = set()

    for raw_line in text.splitlines():
        line = raw_line.strip()
        if not line:
            continue

        parent = None
        child = None

        p_match = re.search(
            rf"parent(?:\s*part(?:\s*number)?)?\s*[:=\-]?\s*({PN_FRAGMENT})",
            line, flags=re.IGNORECASE
        )
        c_match = re.search(
            rf"child(?:\s*part(?:\s*number)?)?\s*[:=\-]?\s*({PN_FRAGMENT})",
            line, flags=re.IGNORECASE
        )
        if p_match and c_match:
            parent = p_match.group(1).upper()
            child = c_match.group(1).upper()

        if not (parent and child):
            m = re.search(
                rf"({PN_FRAGMENT}).*?(?:used\s+in|where\s+used).*?({PN_FRAGMENT})",
                line, flags=re.IGNORECASE
            )
            if m:
                child = m.group(1).upper()
                parent = m.group(2).upper()

        if not (parent and child):
            m = re.search(
                rf"used\s*in\s*part\s*number\s*[:=\-]?\s*({PN_FRAGMENT})",
                line, flags=re.IGNORECASE
            )
            if m:
                parent = m.group(1).upper()
                pns = extract_part_numbers(line)
                for pn in pns:
                    if pn != parent:
                        child = pn
                        break

        if parent and child:
            pair = (parent, child)
            if pair not in seen:
                seen.add(pair)
                rels.append(pair)

    return rels


def build_bom_context(text: str) -> str:
    rels = extract_bom_relationships(text)
    if not rels:
        return ""
    lines = ["Detected BOM context (Used in Part Number = Parent):"]
    for i, (parent, child) in enumerate(rels, start=1):
        lines.append(f"{i}. Parent Part Number: {parent}; Child Part Number: {child}; Used in Part Number: {parent}")
    return "\n".join(lines)


def extract_replacement_pairs(text: str) -> List[Tuple[str, str]]:
    """
    Returns (obsolete_part, replacement_part) pairs inferred from text.
    """
    if not text:
        return []

    pairs: List[Tuple[str, str]] = []
    seen: set[Tuple[str, str]] = set()

    patterns: List[Tuple[str, int, int]] = [
        # "<old> replaced by <new>"
        (rf"({PN_FRAGMENT})\s*(?:is\s*)?(?:to\s*be\s*)?replaced\s*by\s*({PN_FRAGMENT})", 1, 2),
        # "replace <old> with <new>"
        (rf"replace\s*({PN_FRAGMENT})\s*with\s*({PN_FRAGMENT})", 1, 2),
        # "obsolete <old> ... replacement <new>"
        (rf"obsolete(?:d)?\s*[:=\-]?\s*({PN_FRAGMENT}).*?replacement\s*[:=\-]?\s*({PN_FRAGMENT})", 1, 2),
        # "from <old> to <new>"
        (rf"from\s*[:=\-]?\s*({PN_FRAGMENT})\s*to\s*[:=\-]?\s*({PN_FRAGMENT})", 1, 2),
        # "<old> -> <new>" / ">" / unicode arrow
        (rf"({PN_FRAGMENT})\s*(?:->|=>|>|\u2192|\u27a1)\s*({PN_FRAGMENT})", 1, 2),
        # "old applied p/n <old> ... replacement/new p/n <new>"
        (rf"old\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT}).*?(?:replacement|new)\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT})", 1, 2),
        # "replacement/new p/n <new> ... previous/old <old>" (reverse in text)
        (rf"(?:replacement|new)\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT}).*?(?:previous|old)\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT})", 2, 1),
        # Output style support: "Qualified replacement <new> ... Previous <old>"
        (rf"qualified\s+replacement\s+({PN_FRAGMENT}).*?previous\s+({PN_FRAGMENT})", 2, 1),
    ]

    raw_text = str(text)
    for raw_line in raw_text.splitlines():
        line = raw_line.strip()
        if not line:
            continue
        for pat, old_idx, new_idx in patterns:
            m = re.search(pat, line, flags=re.IGNORECASE)
            if not m:
                continue
            old_pn = m.group(old_idx).upper()
            new_pn = m.group(new_idx).upper()
            if old_pn != new_pn:
                pair = (old_pn, new_pn)
                if pair not in seen:
                    seen.add(pair)
                    pairs.append(pair)

    # Multi-line / structured fallback (for JSON-like payload snippets, SPS blocks,
    # and verbose statements where old/new PNs are not on the same line).
    structured_patterns: List[Tuple[str, int, int]] = [
        # JSON payload support: {"obs_part": "<old>", ..., "replacement": "<new>"}
        (
            rf"\"(?:obs_part|obsolete(?:_part)?|old_part|obsolete_part_number)\"\s*:\s*\"({PN_FRAGMENT})\""
            rf".*?\"(?:replacement|replacement_part|new_part|replacement_part_number)\"\s*:\s*\"({PN_FRAGMENT})\"",
            1,
            2,
        ),
        # Labeled multi-line text: "Old Applied P/N: <old> ... Replacement Part Number: <new>"
        (
            rf"old\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT})"
            rf".*?(?:qualified\s+)?replacement\s+(?:part\s+number|p/?n)?\s*[:=\-]?\s*({PN_FRAGMENT})",
            1,
            2,
        ),
        # Reverse-labeled multi-line text
        (
            rf"(?:qualified\s+)?replacement\s+(?:part\s+number|p/?n)?\s*[:=\-]?\s*({PN_FRAGMENT})"
            rf".*?(?:previous|old)\s+(?:applied\s+)?p/?n\s*[:=\-]?\s*({PN_FRAGMENT})",
            2,
            1,
        ),
    ]
    for pat, old_idx, new_idx in structured_patterns:
        for m in re.finditer(pat, raw_text, flags=re.IGNORECASE | re.DOTALL):
            old_pn = m.group(old_idx).upper()
            new_pn = m.group(new_idx).upper()
            if old_pn != new_pn:
                pair = (old_pn, new_pn)
                if pair not in seen:
                    seen.add(pair)
                    pairs.append(pair)

    return pairs


def _enforce_single_blank_line_between_sections(text: str) -> str:
    """Ensure exactly one blank line between recognized section headers/blocks."""
    if not text:
        return ""

    # 1) Normalize line endings and collapse repeated blank lines to one.
    lines = str(text).replace("\r\n", "\n").replace("\r", "\n").split("\n")
    compact: List[str] = []
    prev_blank = False
    for ln in lines:
        is_blank = not ln.strip()
        if is_blank:
            if not prev_blank:
                compact.append("")
            prev_blank = True
        else:
            compact.append(ln.rstrip())
            prev_blank = False

    header_prefixes = {
        "title",
        "problem statement",
        "change drivers",
        "esw turn around",
        "problem description",
        "impact details",
        "issue part number(s)",
        "solution statement",
        "proposed solution",
        "solution description",
        "benefits of the proposed solution",
    }

    def _is_section_header(line: str) -> bool:
        s = re.sub(r"^\s*[-*\u2022]\s*", "", line).strip()
        if not s.endswith(":"):
            return False
        base = s[:-1].strip().lower()
        # Support dynamic headers such as "Change Drivers (PCR# ..., Project# ...):"
        base = re.sub(r"\s*\([^)]*\)\s*$", "", base).strip()
        return base in header_prefixes

    # 2) Ensure one blank line before every section header except first non-empty line.
    out: List[str] = []
    seen_content = False
    for ln in compact:
        if _is_section_header(ln) and seen_content:
            if out and out[-1] != "":
                out.append("")
            elif len(out) >= 2 and out[-1] == "" and out[-2] == "":
                out.pop()
        out.append(ln)
        if ln.strip():
            seen_content = True

    # Final safety: remove accidental double blanks reintroduced by header handling.
    final_lines: List[str] = []
    prev_blank = False
    for ln in out:
        if not ln.strip():
            if not prev_blank:
                final_lines.append("")
            prev_blank = True
        else:
            final_lines.append(ln)
            prev_blank = False

    return "\n".join(final_lines).strip()


def build_replacement_context(text: str) -> str:
    pairs = extract_replacement_pairs(text)
    if not pairs:
        return ""
    lines = ["Detected replacement context (Obsolete -> Replacement):"]
    for i, (old_pn, new_pn) in enumerate(pairs, start=1):
        lines.append(f"{i}. Obsolete Part Number: {old_pn}; Replacement Part Number: {new_pn}")
    return "\n".join(lines)


def clean_ai_output(text: str) -> str:
    if not text:
        return ""
    # Remove markdown code fences that may wrap the entire output
    text = re.sub(r"^```[a-zA-Z]*\n?", "", text, flags=re.MULTILINE)
    text = re.sub(r"^```\s*$", "", text, flags=re.MULTILINE)
    text = re.sub(r"!\[[^\]]*\]\([^)]+\)", "", text)
    text = re.sub(r"<img[^>]*>", "", text, flags=re.IGNORECASE)
    # Strip markdown heading markers (##, ###, etc.) from section headers
    text = re.sub(r"^(\s*)#{1,6}\s+", r"\1", text, flags=re.MULTILINE)
    # Strip bold/italic markers around section header names
    # e.g. **Title:** -> Title:  or **Title**:  -> Title:
    text = re.sub(r"\*{1,3}([^*\n]+?)\*{1,3}", r"\1", text)
    text = re.sub(r"[ \t]+", " ", text)
    text = re.sub(r"\n{3,}", "\n\n", text)
    return text.strip()


def remove_unwanted_sections(text: str) -> str:
    blocked = ["Part Numbers Identified", "BOM Relationships Identified", "Quality Gate Warnings"]
    for header in blocked:
        pattern = rf"\n*{re.escape(header)}\s*:\s*(.*?)(?=\n[A-Za-z][A-Za-z ()/\-]*:\s*|\Z)"
        text = re.sub(pattern, "", text, flags=re.IGNORECASE | re.DOTALL)
    return text.strip()


def _extract_section(text: str, section_name: str) -> str:
    others = [h for h in TOP_LEVEL_SECTION_NAMES if h.lower() != section_name.lower()]
    boundary = "|".join(re.escape(h) for h in others)
    pattern = rf"^\s*(?:[-*\u2022]\s*)?{re.escape(section_name)}\s*:\s*(.*?)(?=^\s*(?:[-*\u2022]\s*)?(?:{boundary})\s*:|\Z)"
    m = re.search(pattern, text, flags=re.IGNORECASE | re.MULTILINE | re.DOTALL)
    return m.group(1).strip() if m else ""


def _extract_subsection(section_text: str, subsection_name: str, valid_names: List[str]) -> str:
    others = [n for n in valid_names if n.lower() != subsection_name.lower()]
    boundary = "|".join(re.escape(n) for n in others)
    pattern = rf"^\s*(?:[-*\u2022]\s*)?{re.escape(subsection_name)}\s*:\s*(.*?)(?=^\s*(?:[-*\u2022]\s*)?(?:{boundary})\s*:|\Z)"
    m = re.search(pattern, section_text, flags=re.IGNORECASE | re.MULTILINE | re.DOTALL)
    return m.group(1).strip() if m else ""


def _extract_title(text: str) -> str:
    m = re.search(r"^\s*Title\s*:\s*(.+)$", text, flags=re.IGNORECASE | re.MULTILINE)
    return m.group(1).strip() if m else ""


def _has_bullet_lines(text: str) -> bool:
    return bool(re.search(r"^\s*[-*•]\s+\S+", text or "", flags=re.MULTILINE))


def _compress_repeated_part_number_labels(text: str) -> str:
    """
    Converts:
      Part Number: 0150-43377 and Part Number: 0150-43378
    into:
      Part Number: 0150-43377, 0150-43378
    """
    if not text:
        return text

    pattern = rf"Part\s*Number\s*:\s*({PN_FRAGMENT})\s*(?:,|and|/)\s*Part\s*Number\s*:\s*({PN_FRAGMENT})"
    prev = None
    cur = text
    while prev != cur:
        prev = cur
        cur = re.sub(pattern, r"Part Number: \1, \2", cur, flags=re.IGNORECASE)
    return cur


def _short_crisp_bullets(text: str, max_bullets: int = 3, max_words: int = 15) -> str:
    if not text:
        return ""
    bullets: List[str] = []
    for raw in text.splitlines():
        s = raw.strip()
        if not s:
            continue
        s = re.sub(r"^[-*•]\s*", "", s).strip()
        words = s.split()
        if len(words) > max_words:
            s = " ".join(words[:max_words]).rstrip(",.;:") + "."
        bullets.append(f"- {s}")
        if len(bullets) >= max_bullets:
            break
    return "\n".join(_dedupe_lines(bullets)).strip()


def _normalize_part_scope_lines(text: str) -> str:
    """
    For Issue Part Number(s):
    - no repeated 'Part Number:' labels
    - OMS documents (0251-xxxxx) are listed one per line with 'OMS' prefix
    - page numbers in brackets are preserved for OMS items
    - if multiple non-OMS PNs in one line, keep as comma-separated PN list only
    """
    if not text:
        return ""

    out: List[str] = []
    for raw in text.splitlines():
        line = re.sub(r"^[-*•]\s*", "", raw.strip())
        if not line:
            continue

        # Detect OMS items (with or without "OMS" prefix, with optional page numbers)
        oms_with_pages = re.findall(
            r"OMS\s+0251-\d{5}(?:\s*\(Pages?\s*[\d\s\-–]+\))?",
            line, flags=re.IGNORECASE
        )
        # Bare 0251-xxxxx numbers not already covered by an "OMS " prefix
        bare_oms = [
            pn for pn in re.findall(r"\b0251-\d{5}\b", line, flags=re.IGNORECASE)
            if not re.search(rf"OMS\s+{re.escape(pn)}", line, flags=re.IGNORECASE)
        ]

        if oms_with_pages or bare_oms:
            # Each OMS document gets its own line with the "OMS" prefix
            # Page numbers are stripped from Part Number sections (they belong in Solution Description)
            for tok in oms_with_pages:
                tok = re.sub(r"\s+", " ", tok).strip()
                # Strip any trailing (Page...) references
                tok = re.sub(r"\s*\(Pages?\s*[\d\s\-\u2013\u2014]+\)", "", tok, flags=re.IGNORECASE).strip()
                if not re.match(r"^OMS\s+", tok, flags=re.IGNORECASE):
                    tok = "OMS " + tok
                out.append(tok)
            for pn in bare_oms:
                out.append(f"OMS {pn.upper()}")
        else:
            pns = extract_part_numbers(line)
            if not pns:
                # Skip non-AM identifiers (e.g., SPS/PCR IDs or supplier part numbers).
                continue

            uniq = []
            seen = set()
            for pn in pns:
                if pn not in seen:
                    seen.add(pn)
                    uniq.append(pn)

            out.append(", ".join(uniq))

    return "\n".join(_dedupe_lines(out)).strip()


def _normalize_multiline_list(text: str) -> str:
    if not text:
        return ""
    if extract_part_numbers(text):
        return _normalize_part_scope_lines(text)

    lines = []
    for raw in text.splitlines():
        ln = re.sub(r"^[-*•]\s*", "", raw.strip())
        if ln:
            lines.append(ln)
    return "\n".join(_dedupe_lines(lines)).strip()


def _linebreak_after_fullstop(text: str) -> str:
    """Put each sentence on a new line using full stop as separator."""
    if not text:
        return ""

    normalized = re.sub(r"\s+", " ", str(text).strip())
    if not normalized:
        return ""

    parts = [p.strip() for p in re.split(r"(?<=\.)\s+", normalized) if p.strip()]
    return "\n".join(parts).strip()


def _extract_revision_number(text: str) -> int | None:
    """Extract revision number from text like Rev 01 / REV.02 / Rev-3."""
    if not text:
        return None
    m = re.search(r"\brev\.?\s*[-:]?\s*0*(\d+)\b", str(text), flags=re.IGNORECASE)
    if not m:
        return None
    try:
        return int(m.group(1))
    except ValueError:
        return None


def _ensure_from_to_revision_order(from_val: str, to_val: str) -> tuple[str, str]:
    """Ensure revision transitions are old->new (e.g., Rev 01 -> Rev 02)."""
    from_rev = _extract_revision_number(from_val)
    to_rev = _extract_revision_number(to_val)

    # Only auto-swap when both sides clearly contain revision numbers.
    if from_rev is not None and to_rev is not None and from_rev > to_rev:
        return to_val, from_val
    return from_val, to_val


def _extract_revision_sequence_from_sps(payload: Dict[str, Any]) -> tuple[int, int] | None:
    """Extract older and newer revision numbers from SPS records to establish From/To order."""
    sps_records = (payload or {}).get("sps_records") or []
    if not sps_records:
        return None
    
    revisions = []
    for rec in sps_records:
        problem = str(rec.get("problem") or "")
        solution = str(rec.get("solution") or "")
        combined = f"{problem} {solution}"
        
        for match in re.finditer(r"\brev\.?\s*[-:]?\s*0*(\d+)", combined, flags=re.IGNORECASE):
            try:
                rev_num = int(match.group(1))
                if rev_num not in revisions:
                    revisions.append(rev_num)
            except ValueError:
                continue
    
    if len(revisions) >= 2:
        revisions.sort()
        return (revisions[0], revisions[-1])
    return None


def _apply_source_based_from_to_order(text: str, payload: Dict[str, Any]) -> str:
    """Apply From/To ordering based on SPS revision sequence if available."""
    if not text or not payload:
        return text
    
    rev_range = _extract_revision_sequence_from_sps(payload)
    if not rev_range:
        return text
    
    old_rev, new_rev = rev_range
    
    # Find all From: ... To: ... patterns and check if they need swapping
    def check_and_swap(m: re.Match) -> str:
        indent = m.group(1)
        from_val = m.group(2).strip()
        to_val = m.group(3).strip()
        
        from_rev = _extract_revision_number(from_val)
        to_rev = _extract_revision_number(to_val)
        
        # If we detected old_rev and new_rev from source, enforce that order
        if from_rev == new_rev and to_rev == old_rev:
            from_val, to_val = to_val, from_val
        
        return f"{indent}From: {from_val}\n{indent}To   : {to_val}"
    
    text = re.sub(
        r'^(\s*)From\s*:\s*(.+?)\s*To\s*:\s*(.+?)\s*$',
        check_and_swap,
        text,
        flags=re.IGNORECASE | re.MULTILINE
    )
    return text


def _format_from_to_changes(text: str) -> str:
    """
    Normalizes From/To change descriptions to a clean two-line format:
        From: <previous value>
        To   : <new value>

    Handles:
      - "From: X, To: Y" or "From: X; To: Y" on one line  → split to two lines
      - "From: X\nTo: Y" (no spacing)                      → align "To   :"
    """
    if not text:
        return text

    result_lines = []
    for line in text.splitlines():
        # Case 1: "From: X, To: Y" on one line — split them
        m1 = re.match(
            r'^(\s*)From\s*:\s*(.+?)\s*[,;]\s*To\s*:\s*(.+?)\s*$',
            line, flags=re.IGNORECASE
        )
        if m1:
            indent = m1.group(1)
            from_val, to_val = _ensure_from_to_revision_order(m1.group(2).strip(), m1.group(3).strip())
            result_lines.append(f"{indent}From: {from_val}")
            result_lines.append(f"{indent}To   : {to_val}")
            continue

        # Case 2: standalone "To: X" immediately after a "From:" line — normalise alignment
        m2 = re.match(r'^(\s*)To\s*:\s*(.+?)\s*$', line, flags=re.IGNORECASE)
        if m2 and result_lines and re.match(r'^\s*From\s*:', result_lines[-1], flags=re.IGNORECASE):
            prev_from = re.match(r'^(\s*)From\s*:\s*(.+?)\s*$', result_lines[-1], flags=re.IGNORECASE)
            if prev_from:
                indent = prev_from.group(1)
                from_val, to_val = _ensure_from_to_revision_order(prev_from.group(2).strip(), m2.group(2).strip())
                result_lines[-1] = f"{indent}From: {from_val}"
                result_lines.append(f"{indent}To   : {to_val}")
            else:
                indent = m2.group(1)
                result_lines.append(f"{indent}To   : {m2.group(2).strip()}")
            continue

        result_lines.append(line)

    return "\n".join(result_lines)


def _bold_from_to_values(text: str) -> str:
    """
    Apply **bold** markers to From: and To: labels and their adjacent values.

    Called as a final post-processing step (after all LLM calls and clean_ai_output runs)
    so the bold markers are never stripped.

    Handles three patterns:
    1. Inline 'From: <value> To: <value>' within a sentence (Problem Description bullets).
    2. Standalone 'From: <value>' line (Solution Description multiline format).
    3. Standalone 'To: <value>' or 'To   : <value>' label line (produced by _format_from_to_changes).
    """
    if not text:
        return text

    # Case 1: Inline "From: <value> To: <value>" on one line / in one sentence.
    # "from" value uses lazy match (.+?) stopping before " To:".
    # "to" value uses greedy non-newline match stopping at sentence terminator.
    text = re.sub(
        r'(?im)\bFrom\s*:\s*(.+?)\s+To\s*:\s*([^.;,\n]+)',
        lambda m: f'**From:** {m.group(1).strip()} **To:** {m.group(2).strip()}',
        text,
    )

    # Case 2: Standalone "From: <value>" line (not already bolded by Case 1).
    # Matches lines that start with optional whitespace then "From:".
    text = re.sub(r'(?im)^(\s*)From\s*:\s*(.+?)$', r'\1**From:** \2', text)

    # Case 3: Standalone "To: <value>" or "To   : <value>" label line.
    # "To\s*:" requires a colon after "To", distinguishing it from "To update..." verbs.
    text = re.sub(r'(?im)^(\s*)To\s*:\s*(.+?)$', r'\1**To:** \2', text)

    return text


def _inline_problem_description_from_to(text: str) -> str:
    """
    Ensures From/To change text is part of the Problem Description sentence,
    not a trailing standalone line.

    Example desired output:
      ... has been changed From: White PVC U:94V-0 To: White PVC FM 4910.
    """
    if not text:
        return text

    lines = [ln.strip() for ln in str(text).splitlines() if ln.strip()]
    if not lines:
        return ""

    out: List[str] = []
    i = 0
    while i < len(lines):
        line = lines[i]

        # Case 1: single-line "From: X To: Y"
        m_inline = re.match(
            r"^From\s*:\s*(.+?)\s+To\s*:\s*(.+?)\s*\.?$",
            line,
            flags=re.IGNORECASE,
        )
        if m_inline:
            from_val, to_val = _ensure_from_to_revision_order(m_inline.group(1).strip(), m_inline.group(2).strip())
            merged = f"From: {from_val} To: {to_val}"
            if out:
                out[-1] = out[-1].rstrip(" .") + f" {merged}."
            else:
                out.append(merged + ".")
            i += 1
            continue

        # Case 2: two-line pair "From: X" then "To: Y"
        m_from = re.match(r"^From\s*:\s*(.+?)\s*\.?$", line, flags=re.IGNORECASE)
        m_to = None
        if m_from and i + 1 < len(lines):
            m_to = re.match(r"^To\s*:\s*(.+?)\s*\.?$", lines[i + 1], flags=re.IGNORECASE)

        if m_from and m_to:
            from_val, to_val = _ensure_from_to_revision_order(m_from.group(1).strip(), m_to.group(1).strip())
            merged = f"From: {from_val} To: {to_val}"
            if out:
                out[-1] = out[-1].rstrip(" .") + f" {merged}."
            else:
                out.append(merged + ".")
            i += 2
            continue

        out.append(line)
        i += 1

    return "\n".join(out).strip()


def _remove_supplier_rev_tokens(text: str) -> str:
    """Remove supplier revision number mentions like Rev.02, Rev 01, rev 01."""
    if not text:
        return ""

    raw = str(text)
    # Remove tokens like Rev.02 / Rev 02 / rev 01 / REV-03 / REV: 04
    raw = re.sub(r"\bREV\.?\s*[-:]?\s*\d+\b", "", raw, flags=re.IGNORECASE)

    # Normalize spacing per-line to preserve section/newline formatting.
    lines = raw.splitlines()
    cleaned_lines: List[str] = []
    for ln in lines:
        ln = re.sub(r"[ \t]{2,}", " ", ln)
        ln = re.sub(r"\s+([,.;:])", r"\1", ln)
        ln = re.sub(r"([,.;:]){2,}", r"\1", ln)
        cleaned_lines.append(ln.rstrip())

    # Keep intentional blank lines between sections, collapse only excessive empties.
    out = "\n".join(cleaned_lines)
    out = re.sub(r"\n{3,}", "\n\n", out)
    return out.strip()


def _translate_sps_to_professional(text: str) -> str:
    """
    Converts raw supplier-voice SPS text to third-person professional language
    suitable for Applied Materials engineering documentation.
    Handles common first-person / informal phrases.
    """
    if not text:
        return text

    t = str(text)

    # First-person → third-person supplier references
    replacements = [
        # "we are unable" / "we cannot"
        (r"\bwe are unable to\b",        "Supplier is unable to"),
        (r"\bwe cannot\b",               "Supplier cannot"),
        (r"\bwe can not\b",              "Supplier cannot"),
        (r"\bwe are not able to\b",      "Supplier is unable to"),
        # "we are able" / "we can"
        (r"\bwe are able to\b",          "Supplier is able to"),
        (r"\bwe can\b",                  "Supplier can"),
        # "we have" / "we had" / "we will"
        (r"\bwe have\b",                 "Supplier has"),
        (r"\bwe had\b",                  "Supplier had"),
        (r"\bwe will\b",                 "Supplier will"),
        (r"\bwe would\b",                "Supplier would"),
        # "we are" general
        (r"\bwe are\b",                  "Supplier is"),
        # standalone "we" as subject
        (r"\bwe\b",                      "the Supplier"),
        # "our" possessive
        (r"\bour\b",                     "Supplier\'s"),
        # "us" object
        (r"\bus\b",                      "the Supplier"),
        # informal / vague phrases
        (r"\bplease\b",                  ""),
        (r"\bkindly\b",                  ""),
        (r"\bFYI\b",                     ""),
        (r"\bfor your information\b",    ""),
    ]

    for pattern, replacement in replacements:
        t = re.sub(pattern, replacement, t, flags=re.IGNORECASE)

    # Collapse any double-spaces introduced by blank replacements
    t = re.sub(r"  +", " ", t).strip(" ,;.")
    return t


def _to_single_line_summary(text: str, max_chars: int = 220) -> str:
    if not text:
        return ""

    normalized = re.sub(r"\s+", " ", str(text).strip())
    if not normalized:
        return ""

    sentence_parts = re.split(r"(?<=[.!?])\s+", normalized)
    single_line = sentence_parts[0].strip() if sentence_parts else normalized
    if len(single_line) > max_chars:
        single_line = single_line[: max_chars - 3].rstrip(" ,;:-") + "..."
    return single_line


def _build_reference_change_summary_from_payload(payload: Dict[str, Any]) -> str:
    """
    Builds the Change Drivers section content (no header - header is added by pipeline).
    PCR + Project records are combined into TWO concise bullets.
    ESW records are NOT included here - they appear only in the ESW Turn Around section.
    """
    pcr_records = payload.get("pcr_records") or []
    project_records = payload.get("project_records") or []

    # SPS and ESW records are intentionally excluded from the Change Drivers.

    pcr_project_chunks: List[str] = []
    for rec in pcr_records:
        pcr_id = rec.get("pcr_id", "")
        problem = _to_single_line_summary(rec.get("problem") or "")
        solution = _to_single_line_summary(rec.get("solution") or "")
        if pcr_id:
            pcr_project_chunks.append(
                f"PCR {pcr_id}:\n  Problem: {problem}\n  Solution: {solution}"
            )
    for rec in project_records:
        project_id = rec.get("project_id", "")
        scope = _to_single_line_summary(rec.get("defined_scope") or rec.get("project_name") or "")
        deliverables = _to_single_line_summary(rec.get("deliverables") or "")
        if project_id:
            pcr_project_chunks.append(
                f"Project {project_id}:\n  Scope: {scope}\n  Deliverables: {deliverables}"
            )

    if not pcr_project_chunks:
        return ""

    combined_input = "\n\n".join(pcr_project_chunks)
    prompt = (
        "You are a senior manufacturing engineer summarizing Engineering Change Records.\n\n"
        "Read ALL of the following PCR and Project records together and write exactly 2 bullets "
        "that provide a combined high-level summary of what all records are collectively driving.\n\n"
        "Rules:\n"
        "- Each bullet must start with 'This change is to drive'.\n"
        "- Each bullet must be a complete sentence with maximum 15 words.\n"
        "- Do not write separate bullets per record; synthesize across all records.\n"
        "- Use plain professional language. Do NOT use arrows, special symbols, or SPS numbers.\n"
        "- Output ONLY the 2 bullet lines starting with '- '. No preamble or explanation.\n\n"
        "INPUT RECORDS:\n" + combined_input
    )
    result = _call_databricks(prompt)
    # If LLM returns empty, retry once if there are PCR/Project records
    if (not result or result.strip() == "") and pcr_project_chunks:
        result = _call_databricks(prompt)
    if result and not result.startswith("Error:") and result.strip():
        return result.strip()
    # Fallback: plain reference lines
    fallback = [
        "- This change is to drive aligned PCR and Project scope for controlled implementation.",
        "- This change is to drive documented execution readiness across impacted records.",
    ]
    return "\n".join(fallback)


def _build_esw_turnaround_from_payload(payload: Dict[str, Any]) -> str:
    """
    Builds the 'ESW Turn Around (ESW# XXXXX):' section.
    One header per ESW with 1-2 LLM-generated bullets describing temporary workaround context.
    ESW number appears in the section header, not inside the bullets.
    Include issue part numbers in bullets when they can be inferred from ESW text.
    """
    esw_records = payload.get("esw_records") or []
    if not esw_records:
        return ""

    blocks: List[str] = []
    for rec in esw_records:
        ec_number = rec.get("ec_number") or rec.get("esw_number", "")
        if not ec_number:
            continue
        title = _to_single_line_summary(rec.get("title") or "")
        problem = _to_single_line_summary(rec.get("problem") or "")
        solution = _to_single_line_summary(rec.get("solution") or "")
        esw_source_text = f"{title}\n{problem}\n{solution}"
        esw_part_numbers = _unique_preserve(extract_part_numbers(esw_source_text))
        part_hint = ", ".join(esw_part_numbers[:3])
        part_hint_line = f"\n  Part Numbers: {part_hint}" if part_hint else ""
        esw_input = (
            f"Title: {title}\n"
            f"  Problem: {problem}\n"
            f"  Solution: {solution}"
            f"{part_hint_line}"
        )
        prompt = (
            "You are a senior manufacturing engineer.\n\n"
            "Read the following ESW record and write 1 or 2 bullet points that briefly describe "
            "how the issue is being temporarily addressed through this ESW (interim/workaround solution).\n\n"
            "Rules:\n"
            "- Do NOT mention the ESW number in the bullets (it is already in the section header).\n"
            "- Mention Applied part number(s) in the bullets when provided in the ESW record.\n"
            "- Each bullet must be a complete sentence of maximum 15 words.\n"
            "- Use plain, concise, third-person professional language.\n"
            "- Do NOT use arrows, special symbols, or SPS numbers.\n"
            "- Output ONLY 1 or 2 bullet lines starting with '- '. No preamble, no explanation.\n\n"
            "ESW RECORD:\n" + esw_input
        )
        result = _call_databricks(prompt)
        # If LLM returns empty, retry once
        if not result or result.strip() == "":
            result = _call_databricks(prompt)
        if result and not result.startswith("Error:") and result.strip():
            bullet_text = result.strip()
        else:
            if part_hint:
                bullet_text = (
                    f"- Temporary ESW workaround is active for Applied P/N {part_hint} "
                    "pending final ECR implementation."
                )
            else:
                bullet_text = "- Temporary ESW workaround is in place pending final ECR implementation."
        blocks.append(f"ESW Turn Around (ESW# {ec_number}):\n{bullet_text}")

    return "\n\n".join(blocks).strip()


def _split_inline_oms_list(text: str) -> str:
    """
    Splits inline OMS document lists (joined with ' - ') into one OMS entry per line.
    Preserves page numbers in brackets when present.

    OMS documents (0251-xxxxx) are child parts that report to Kit (0241-xxxxx)
    or Assembly (0010-xxxxx / 0011-xxxxx) parent part numbers.

    Example:
      IN:  OMS 0251-18026 - OMS 0251-23910 - OMS 0251-00319
      OUT: OMS 0251-18026
           OMS 0251-23910
           OMS 0251-00319

      IN:  OMS 0251-18026 (Page 5) - OMS 0251-23910 (Pages 3-4)
      OUT: OMS 0251-18026 (Page 5)
           OMS 0251-23910 (Pages 3-4)
    """
    if not text:
        return text

    out_lines: List[str] = []
    for raw_line in text.splitlines():
        # Only process lines that contain 2 or more OMS items
        oms_count = len(re.findall(r"OMS\s+0251-\d{5}", raw_line, flags=re.IGNORECASE))
        if oms_count < 2:
            out_lines.append(raw_line)
            continue

        # Preserve indentation and leading bullet marker
        indent = raw_line[: len(raw_line) - len(raw_line.lstrip())]
        stripped = raw_line.lstrip()
        bullet_match = re.match(r"^([-*•]\s*)", stripped)
        bullet = bullet_match.group(1) if bullet_match else ""
        content = stripped[len(bullet):].strip()

        # Detect any intro text before the first OMS item
        first_oms_match = re.search(r"OMS\s+0251-\d{5}", content, flags=re.IGNORECASE)
        intro = content[: first_oms_match.start()].rstrip(" :-") if first_oms_match else ""

        # Extract all OMS items with their optional page references
        oms_items = re.findall(
            r"OMS\s+0251-\d{5}(?:\s*\(Pages?\s*[\d\s\-–]+\))?",
            content, flags=re.IGNORECASE
        )
        oms_items = [re.sub(r"\s+", " ", item).strip() for item in oms_items]

        if intro:
            # Intro text becomes a header line; OMS items become indented sub-bullets
            out_lines.append(f"{indent}{bullet}{intro}:")
            for item in oms_items:
                out_lines.append(f"{indent}  - {item}")
        else:
            # No intro: each OMS item on its own line preserving original indent/bullet
            for item in oms_items:
                out_lines.append(f"{indent}{bullet}{item}")

    return "\n".join(out_lines)


def _normalize_bullets(text: str, crisp: bool = False) -> str:
    if not text:
        return ""
    if crisp:
        return _short_crisp_bullets(text, max_bullets=3, max_words=15)

    # Split any inline OMS lists (e.g. "OMS 0251-A - OMS 0251-B") before normalizing.
    text = _split_inline_oms_list(text)

    # Normalize mixed numbering/bullet styles into clean bullet points.
    # Indented sub-bullet lines (2+ leading spaces) are preserved so that
    # OMS document sub-bullets with page numbers are never flattened.
    raw = str(text)
    raw = re.sub(r"\s+(?=\d+\.\s+)", "\n", raw)

    lines = []
    for raw_line in raw.splitlines():
        # Preserve indented sub-bullets (2+ leading spaces) — these are OMS/doc sub-items.
        if re.match(r"^  +", raw_line):
            content = raw_line.rstrip()
            stripped_content = content.lstrip()
            indent = content[: len(content) - len(stripped_content)]
            stripped_content = re.sub(r"^[-*•]\s*", "", stripped_content)
            stripped_content = re.sub(r"\s+", " ", stripped_content).strip()
            if stripped_content:
                lines.append(f"{indent}- {stripped_content}")
            continue

        ln = raw_line.strip()
        ln = re.sub(r"^[-*•]\s*", "", ln)
        ln = re.sub(r"^\(?\d+\)?[.)]\s*", "", ln)
        ln = re.sub(r"^[A-Za-z][.)]\s*", "", ln)
        if re.fullmatch(r"\d+\.?", ln):
            continue
        ln = re.sub(r"\s+", " ", ln).strip(" -–—:;,")
        if ln:
            # Allow trailing ":" for grouped-bullet headers (e.g. "Update OMS docs:").
            if not re.search(r"[.!?:]$", ln):
                ln = ln + "."
            lines.append(f"- {ln}")
    return "\n".join(_dedupe_lines(lines)).strip()


def _extract_scope_part_descriptions(source_text: str) -> Dict[str, str]:
    """
    Parses 'Scope Parts:' block in source_text and returns a mapping of
    part_number -> part_description for entries like '0150-43377 - RING, FOCUS'.
    """
    pn_desc: Dict[str, str] = {}
    in_scope = False
    for raw in (source_text or "").splitlines():
        line = raw.strip()
        if re.match(r"^Scope\s+Parts\s*:", line, flags=re.IGNORECASE):
            in_scope = True
            continue
        if in_scope:
            if not line:
                continue
            # Stop if we hit another top-level section header
            if re.match(r"^[A-Za-z][A-Za-z\s]+:", line) and not line.startswith("-"):
                break
            entry = re.sub(r"^[-*•]\s*", "", line).strip()
            pns = extract_part_numbers(entry)
            if pns:
                pn = pns[0]
                desc = re.sub(rf"\b{re.escape(pn)}\b", "", entry, flags=re.IGNORECASE)
                desc = re.sub(r"\s+", " ", desc).strip(" -–—:;,.")
                if desc:
                    pn_desc[pn] = desc
    return pn_desc


def _extract_labeled_part_descriptions(source_text: str) -> Dict[str, str]:
    """Parse labeled part fields like 'Part Number:' and 'Part Description:' from input text."""
    pn_desc: Dict[str, str] = {}
    current_pn = ""

    for raw in (source_text or "").splitlines():
        line = raw.strip()
        if not line:
            continue

        m_pn = re.search(
            rf"\bPart\s*Number\s*:\s*({PN_FRAGMENT})\b",
            line,
            flags=re.IGNORECASE,
        )
        if m_pn:
            current_pn = m_pn.group(1).upper()

        m_desc = re.search(r"\bPart\s*Description\s*:\s*(.+)$", line, flags=re.IGNORECASE)
        if m_desc and current_pn:
            desc = re.sub(r"\s+", " ", m_desc.group(1)).strip(" -:;,.")
            if desc and current_pn not in pn_desc:
                pn_desc[current_pn] = desc

    return pn_desc


def _build_part_description_map(source_text: str) -> Dict[str, str]:
    """Combine known part-description sources for deterministic formatting."""
    out = _extract_scope_part_descriptions(source_text)
    labeled = _extract_labeled_part_descriptions(source_text)
    for pn, desc in labeled.items():
        if pn not in out and desc:
            out[pn] = desc
    return out


def _unique_preserve(items: List[str]) -> List[str]:
    seen = set()
    out: List[str] = []
    for item in items:
        val = str(item or "").strip().upper()
        if val and val not in seen:
            seen.add(val)
            out.append(val)
    return out


def _extract_bom_from_narrative(text: str) -> List[Tuple[str, str]]:
    """
    Parses narrative sentences to extract (parent, child) BOM pairs.
    Handles patterns like:
      - "The BOM of 0051-43804 calls for one unit of 3300-16961"
      - "0051-43804 BOM requires 3300-16961"
      - "3300-16961 is a child of 0051-43804"
      - "0051-43804 contains / includes 3300-16961"
    """
    if not text:
        return []

    rels: List[Tuple[str, str]] = []
    seen: set = set()

    PNF = PN_FRAGMENT  # reuse the existing pattern fragment

    patterns = [
        # "BOM of X calls for / requires / contains Y"
        (rf"(?:BOM\s+of|BOM\s+for)\s+({PNF}).*?(?:calls?\s+for|requires?|contains?|includes?)\s+.*?({PNF})", 1, 2),
        # "X BOM calls for / requires Y"
        (rf"({PNF})\s+BOM\s+.*?(?:calls?\s+for|requires?|contains?|includes?)\s+.*?({PNF})", 1, 2),
        # "X is the parent of Y" / "X is parent part"
        (rf"({PNF})\s+is\s+(?:the\s+)?parent(?:\s+part)?\s+(?:of|for)\s+({PNF})", 1, 2),
        # "Y is a child of X" / "Y is child part of X"
        (rf"({PNF})\s+is\s+(?:a\s+)?child(?:\s+part)?\s+of\s+({PNF})", 2, 1),
        # "Y reports to X"
        (rf"({PNF})\s+reports?\s+to\s+({PNF})", 2, 1),
        # "assembly X uses / contains Y"
        (rf"(?:assembly|assy)\s+({PNF})\s+.*?(?:uses?|contains?|includes?|calls?\s+for)\s+.*?({PNF})", 1, 2),
    ]

    for raw_line in text.splitlines():
        line = raw_line.strip()
        if not line:
            continue
        for pattern, parent_grp, child_grp in patterns:
            m = re.search(pattern, line, flags=re.IGNORECASE)
            if m:
                parent = m.group(parent_grp).upper()
                child = m.group(child_grp).upper()
                if parent != child:
                    pair = (parent, child)
                    if pair not in seen:
                        seen.add(pair)
                        rels.append(pair)
                break  # one match per line is enough

    return rels


def _derive_part_sections_from_bom(
    normalized_affected: str,
    normalized_issue: str,
    source_text: str,
    narrative_text: str = "",
) -> Tuple[str, str]:
    """Ensure Issue part tracks child issue part; Affected part tracks parent part.
    First tries structured BOM context from source_text, then falls back to
    parsing narrative sentences in narrative_text (e.g. Problem Description)."""
    rels = extract_bom_relationships(source_text or "")
    # Fallback: parse narrative text (e.g. generated Problem Description)
    if not rels and narrative_text:
        rels = _extract_bom_from_narrative(narrative_text)
    # Also try narrative in source_text itself
    if not rels:
        rels = _extract_bom_from_narrative(source_text or "")
    if not rels:
        return normalized_affected, normalized_issue

    parent_to_children: Dict[str, List[str]] = {}
    child_to_parents: Dict[str, List[str]] = {}
    for parent, child in rels:
        parent_to_children.setdefault(parent, []).append(child)
        child_to_parents.setdefault(child, []).append(parent)

    affected_pns = _unique_preserve(extract_part_numbers(normalized_affected))
    issue_pns = _unique_preserve(extract_part_numbers(normalized_issue))

    # If Issue is blank, infer it from children of selected parents.
    if not issue_pns and affected_pns:
        inferred_issue: List[str] = []
        for parent in affected_pns:
            inferred_issue.extend(parent_to_children.get(parent, []))
        issue_pns = _unique_preserve(inferred_issue)

    # If Affected is blank, infer it from parents of selected issue parts.
    if not affected_pns and issue_pns:
        inferred_affected: List[str] = []
        for child in issue_pns:
            inferred_affected.extend(child_to_parents.get(child, []))
        affected_pns = _unique_preserve(inferred_affected)

    # If both are blank, derive complete mapping from BOM relationships.
    if not affected_pns and not issue_pns:
        affected_pns = _unique_preserve([p for p, _ in rels])
        issue_pns = _unique_preserve([c for _, c in rels])

    # If Issue already contains mixed values, keep only true child issue parts when BOM context exists.
    if issue_pns:
        direct_children = {c for _, c in rels}
        filtered_issue = [pn for pn in issue_pns if pn in direct_children]
        if filtered_issue:
            issue_pns = _unique_preserve(filtered_issue)

    # When issue part(s) are known, Affected must contain only first-level parents of those issue parts.
    if issue_pns:
        direct_affected: List[str] = []
        for child in issue_pns:
            direct_affected.extend(child_to_parents.get(child, []))
        direct_affected = _unique_preserve(direct_affected)
        if direct_affected:
            affected_pns = direct_affected

    # Guardrail: do not allow parent parts to remain in Issue section.
    if issue_pns and affected_pns:
        affected_set = set(affected_pns)
        filtered_issue = [pn for pn in issue_pns if pn not in affected_set]
        if filtered_issue:
            issue_pns = _unique_preserve(filtered_issue)

    affected_text = ", ".join(affected_pns) if affected_pns else normalized_affected
    issue_text = ", ".join(issue_pns) if issue_pns else normalized_issue
    return affected_text, issue_text


def _dedupe_problem_sentences(text: str, max_lines: int = 0) -> str:
    """Remove near-duplicate lines from Problem Description while preserving concise detail."""
    if not text:
        return ""

    lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
    if not lines:
        return ""

    def _norm(s: str) -> str:
        return re.sub(r"\s+", " ", re.sub(r"[^a-z0-9 ]", " ", s.lower())).strip()

    def _tokens(s: str) -> set:
        return {t for t in _norm(s).split() if t}

    def _looks_duplicate(a: str, b: str) -> bool:
        na, nb = _norm(a), _norm(b)
        if not na or not nb:
            return False
        if na == nb or na in nb or nb in na:
            return True

        ta, tb = _tokens(a), _tokens(b)
        if not ta or not tb:
            return False
        overlap = len(ta & tb) / float(min(len(ta), len(tb)))
        return overlap >= 0.92

    kept: List[str] = []
    for ln in lines:
        if not kept:
            kept.append(ln)
            continue
        if any(_looks_duplicate(existing, ln) for existing in kept):
            continue
        kept.append(ln)
        if max_lines > 0 and len(kept) >= max_lines:
            break

    return "\n".join(kept).strip()


def _dedupe_bullet_block(text: str, max_lines: int = 0) -> str:
    """Remove exact and near-duplicate bullet lines while preserving bullet formatting."""
    if not text:
        return ""

    raw_lines = [ln.rstrip() for ln in text.splitlines() if ln.strip()]
    if not raw_lines:
        return ""

    def _body(line: str) -> str:
        return re.sub(r"^\s*[-*\u2022]\s*", "", line).strip()

    def _norm(s: str) -> str:
        return re.sub(r"\s+", " ", re.sub(r"[^a-z0-9 ]", " ", s.lower())).strip()

    kept: List[str] = []
    kept_norms: List[str] = []
    for ln in raw_lines:
        body = _body(ln)
        norm = _norm(body)
        if not norm:
            continue
        is_dup = False
        for existing in kept_norms:
            if norm == existing or norm in existing or existing in norm:
                is_dup = True
                break
        if is_dup:
            continue
        kept.append(ln if re.match(r"^\s*[-*\u2022]\s+", ln) else f"- {body}")
        kept_norms.append(norm)
        if max_lines > 0 and len(kept) >= max_lines:
            break

    return "\n".join(kept).strip()


def _break_long_sentences(text: str, max_length: int = 180, max_words: int = 30) -> str:
    """Intelligently break overly long, complex sentences into 2-3 shorter, clearer sentences."""
    if not text:
        return ""

    text = str(text).strip()
    
    # If text is short enough, return as-is
    if len(text) <= max_length:
        return text
    
    sentences = re.split(r'(?<=[.!?])\s+', text)
    result_sentences: List[str] = []
    
    for sent in sentences:
        sent = sent.strip()
        if not sent:
            continue
        
        # If sentence is short enough, keep it as-is
        if len(sent) <= max_length:
            result_sentences.append(sent)
            continue
        
        # Try to break long sentence at semicolons or logical break points
        word_count = len(sent.split())
        if word_count <= max_words:
            result_sentences.append(sent)
            continue
        
        # Look for break points
        broken = False
        
        # Try splitting on semicolons
        if ';' in sent:
            parts = [p.strip() for p in sent.split(';')]
            result_sentences.extend([p + '.' if not p.endswith(('.', '!', '?')) else p for p in parts if p])
            broken = True
        
        # Try splitting before common phrases that start secondary clauses
        if not broken:
            break_patterns = [
                r'(\,\s*(?:resulting\s+in|causing|preventing|leading\s+to|thereby|which)\s+)',
                r'(\,\s*(?:however|but|yet|additionally|further|furthermore|moreover)\s+)',
                r'(\s+(?:Currently|Only|Specifically)\s+)',
            ]
            
            for pattern in break_patterns:
                if re.search(pattern, sent, flags=re.IGNORECASE):
                    match = re.search(pattern, sent, flags=re.IGNORECASE)
                    if match:
                        pos = match.start()
                        before = sent[:pos].rstrip(', ')
                        after = sent[pos:].lstrip(' ,')
                        if after.lower().startswith(('resulting', 'causing', 'preventing', 'leading', 'thereby', 'which', 'however', 'but', 'yet')):
                            # Capitalize first letter of second sentence
                            after = after[0].upper() + after[1:] if after else after
                        
                        # Ensure both parts end correctly
                        if before and not before.endswith(('.', '!', '?')):
                            before = before + '.'
                        if after and not after.endswith(('.', '!', '?')):
                            after = after + '.'
                        
                        result_sentences.append(before)
                        result_sentences.append(after)
                        broken = True
                        break
        
        # If no break point found, split after every ~25-30 words
        if not broken:
            words = sent.split()
            chunk_size = max_words - 5
            for i in range(0, len(words), chunk_size):
                chunk = ' '.join(words[i:i+chunk_size])
                if not chunk.endswith(('.', '!', '?')):
                    chunk += '.'
                result_sentences.append(chunk)
            broken = True
        
        if not broken:
            result_sentences.append(sent)
    
    return '\n'.join(result_sentences).strip()


def _inject_single_part_description(part_text: str, source_text: str) -> str:
    """If a part section has one unique part number and we know its description, append it."""
    if not part_text:
        return ""

    pns = extract_part_numbers(part_text)
    unique_pns = list(dict.fromkeys(pns))
    if len(unique_pns) != 1:
        return part_text

    pn = unique_pns[0]
    desc_map = _build_part_description_map(source_text)
    desc = (desc_map.get(pn) or "").strip()
    if not desc:
        return part_text

    return f"{pn} - {desc}"


def _extract_short_title_input(source_text: str) -> str:
    m = re.search(r"^\s*Short\s+Title\s+Input\s*:\s*(.+)$", source_text or "", flags=re.IGNORECASE | re.MULTILINE)
    return m.group(1).strip() if m else ""


def _remove_known_part_descriptions(text: str, source_text: str, remove_standalone: bool = False) -> str:
    """Strip known part descriptions from free-form text while preserving part numbers."""
    if not text:
        return ""

    cleaned = str(text)
    desc_map = _build_part_description_map(source_text)
    if desc_map:
        for pn, desc in sorted(desc_map.items(), key=lambda item: len(item[1]), reverse=True):
            desc = str(desc or "").strip()
            if not desc:
                continue

            pn_pat = re.escape(pn)
            desc_pat = re.escape(desc)

            cleaned = re.sub(
                rf"(\b(?:P/N|PN|Part\s*Number)\s*[:#-]?\s*{pn_pat}\b)\s*(?:[-–—,;:]|\()\s*{desc_pat}\s*(?:\))?",
                r"\1",
                cleaned,
                flags=re.IGNORECASE,
            )
            cleaned = re.sub(
                rf"(\b{pn_pat}\b)\s*(?:[-–—,;:]|\()\s*{desc_pat}\s*(?:\))?",
                r"\1",
                cleaned,
                flags=re.IGNORECASE,
            )

            if remove_standalone:
                cleaned = re.sub(
                    rf"(?<![A-Za-z0-9]){desc_pat}(?![A-Za-z0-9])",
                    "",
                    cleaned,
                    flags=re.IGNORECASE,
                )

    # Generic cleanup: if a bracketed token follows a valid AM part number,
    # treat it as part description and remove it from narrative text.
    cleaned = re.sub(
        rf"(\b{PN_FRAGMENT}\b)\s*\([^)]*\)",
        r"\1",
        cleaned,
        flags=re.IGNORECASE,
    )

    cleaned = re.sub(r"\s+", " ", cleaned)
    cleaned = re.sub(r"\s+([,.;:])", r"\1", cleaned)
    cleaned = re.sub(r"\(\s*\)", "", cleaned)
    cleaned = re.sub(r"\s*[-–—,:;]\s*[-–—,:;]\s*", " - ", cleaned)
    cleaned = re.sub(r"\s{2,}", " ", cleaned)
    return cleaned.strip(" -–—:;,.")


def _derive_complete_title_body(source_text: str) -> str:
    """Build a clearer fallback title body when the model returns a clipped fragment."""
    candidates: List[str] = []

    short_title = _extract_short_title_input(source_text)
    if short_title:
        candidates.append(short_title)

    patterns = [
        r"^\s*Title\s*:\s*(.+)$",
        r"^\s*Key\s+Objective\s*:\s*(.+)$",
        r"^\s*Problem\s+Statement\s+Input\s*:\s*(.+)$",
        r"^\s*Proposed\s+Solution\s+Input\s*:\s*(.+)$",
        r"^\s*Supplier-reported issue .*?:\s*(.+)$",
    ]
    for pattern in patterns:
        candidates.extend(re.findall(pattern, source_text or "", flags=re.IGNORECASE | re.MULTILINE))

    for raw in candidates:
        body = _remove_known_part_descriptions(raw, source_text, remove_standalone=True)
        body = re.sub(r"\bPart\s*Number\b", "", body, flags=re.IGNORECASE)
        body = re.sub(rf"\b{PN_FRAGMENT}\b", "", body, flags=re.IGNORECASE)
        body = re.sub(
            r"^\s*(?:Per\s+SPS#\s*\d+[, ]*)?(?:Supplier\s+notified(?:\s+Applied\s+Materials)?\s+that\s+)?",
            "",
            body,
            flags=re.IGNORECASE,
        )
        body = re.split(r"(?<=[.!?])\s+", body, maxsplit=1)[0]
        body = re.split(
            r"\b(?:resulting\s+in|causing|preventing|leading\s+to|thereby|because|since|so\s+that|currently|only)\b",
            body,
            maxsplit=1,
            flags=re.IGNORECASE,
        )[0]
        body = re.sub(r"\s+", " ", body).strip(" -–—:;,.")
        if body and not _title_looks_incomplete(body):
            return body

    return ""


def _trim_title_words(text: str, max_len: int = 85) -> str:
    t = re.sub(r"\s+", " ", str(text or "")).strip(" -–—:;,.")
    if len(t) <= max_len:
        return t

    words = t.split()
    out = ""
    for w in words:
        candidate = w if not out else f"{out} {w}"
        if len(candidate) > max_len:
            break
        out = candidate
    return out if out else t[:max_len].rstrip(" -–—:;,.")


def _title_looks_incomplete(text: str) -> bool:
    t = str(text or "").strip().lower()
    if not t:
        return True
    # Pure number or single/very-short token with no engineering meaning.
    if re.fullmatch(r"\d+", t):
        return True
    if len(t) < 5:
        return True
    if t.endswith(("-", "/", ":", ";", ",", "(")):
        return True
    if re.search(r"\b(?:and|or|with|for|to)\s*$", t):
        return True
    if re.search(r"\bnote\s+\d+\s*$", t):
        return True
    # Common clipped suffixes seen when text is cut mid-word.
    bad_suffixes = (
        "repla", "replac", "modif", "obsolescen", "specifi", "substitu", "materia", "procuremen",
    )
    return any(t.endswith(sfx) for sfx in bad_suffixes)


def _normalize_title_with_part_number(
    title: str,
    source_text: str,
    lead_pn_override: str = "",
    force_no_prefix: bool = False,
) -> str:
    """
    Enforces title format: <PART_NUMBER> - <short technical title>
    and removes 'Part Number' wording and part descriptions from title.
    """
    t = str(title or "").strip()
    if not t:
        return ""

    title_pns = extract_part_numbers(t)
    source_pns = extract_part_numbers(source_text)
    # Never keep more than one part number in Title.
    lead_pn = (lead_pn_override or "").strip().upper() or (source_pns[0] if source_pns else "")

    # Start by removing ALL PNs from body, then optionally prepend one clean lead PN.
    body = re.sub(rf"\bPart\s*Number\s*[:\-]?\s*{PN_FRAGMENT}\b", "", t, flags=re.IGNORECASE)
    body = re.sub(rf"\b{PN_FRAGMENT}\b", "", body, flags=re.IGNORECASE)
    body = re.sub(r"\bPart\s*Number\b", "", body, flags=re.IGNORECASE)
    body = re.sub(r"\s*[-–—]\s*[-–—]+\s*", " - ", body)
    body = re.sub(r"\s+", " ", body).strip(" -–—:;,.")
    body = _remove_known_part_descriptions(body, source_text, remove_standalone=True)

    # Remove part description (item name) from the title body if it appears there.
    pn_desc_map = _extract_scope_part_descriptions(source_text)
    part_desc = pn_desc_map.get(lead_pn, "")
    if part_desc:
        body = re.sub(
            rf"^{re.escape(part_desc)}\s*[-–—]?\s*",
            "",
            body,
            flags=re.IGNORECASE,
        ).strip(" -–—:;,.")

    # If body is empty, fallback to Short Title input from payload composition.
    if not body:
        body = _extract_short_title_input(source_text)
        body = re.sub(rf"\b{PN_FRAGMENT}\b", "", body, flags=re.IGNORECASE)
        body = re.sub(r"\s+", " ", body).strip(" -–—:;,.")
        body = _remove_known_part_descriptions(body, source_text, remove_standalone=True)

    if not body or _title_looks_incomplete(body):
        fallback_body = _derive_complete_title_body(source_text)
        if fallback_body:
            body = fallback_body

    if not body:
        body = "Engineering Change Update"

    # If the draft had multiple PNs or the sentence looks clipped, do not prefix PN.
    keep_pn_prefix = (not force_no_prefix) and bool(lead_pn) and len(set(title_pns)) <= 1 and not _title_looks_incomplete(body)

    body_limit = max(20, 85 - len(lead_pn) - 3) if keep_pn_prefix else 85
    body = _trim_title_words(body, max_len=body_limit)
    candidate = f"{lead_pn} - {body}" if keep_pn_prefix else body

    # Guard against trimmed clipped-looking result.
    if _title_looks_incomplete(candidate):
        fallback_body = _derive_complete_title_body(source_text) or body
        no_pn = _trim_title_words(fallback_body, max_len=85)
        return no_pn or "Engineering Change Update"

    return candidate or "Engineering Change Update"


def _extract_issue_part_numbers_from_problem(problem_statement: str) -> List[str]:
    """Return ordered unique issue P/Ns from normalized Problem Statement text.

    Priority:
    1) Issue Part Number(s)
    2) Issue Part Number(s) fallback when the Issue section is not present.
    """
    if not problem_statement:
        return []

    issue = _extract_subsection(problem_statement, "Issue Part Number(s)", PROBLEM_SUBSECTION_NAMES)
    if not issue:
        issue = _extract_subsection(problem_statement, "Issue Part Numbers", PROBLEM_SUBSECTION_NAMES)
    if not issue:
        issue = ""

    pns = extract_part_numbers(issue or "")
    seen: set[str] = set()
    out: List[str] = []
    for pn in pns:
        if pn not in seen:
            seen.add(pn)
            out.append(pn)
    return out


def _apply_engineering_shortcuts(text: str) -> str:
    """
    Replaces known engineering full terms with approved shortcuts.
    """
    if not text:
        return ""

    out = str(text)
    for full_term, shortcut in sorted(ENGINEERING_SHORTCUTS, key=lambda x: len(x[0]), reverse=True):
        pattern = rf"(?<!\w){re.escape(full_term)}(?!\w)"
        out = re.sub(pattern, shortcut, out, flags=re.IGNORECASE)
    return out


# ----------------------------
# Validation
# ----------------------------
def validate_engineering_output(output: str, source_text: str) -> List[str]:
    violations: List[str] = []

    if not re.search(r"^\s*Title\s*:", output, flags=re.IGNORECASE | re.MULTILINE):
        violations.append("Missing 'Title:' line.")
    if not re.search(r"^\s*Problem Statement\s*:", output, flags=re.IGNORECASE | re.MULTILINE):
        violations.append("Missing 'Problem Statement:' section.")
    if not re.search(r"^\s*(Solution Statement|Proposed Solution)\s*:", output, flags=re.IGNORECASE | re.MULTILINE):
        violations.append("Missing 'Solution Statement:' section.")

    title = _extract_title(output)
    if not title:
        violations.append("Title must not be empty.")

    if re.search(FIRST_PERSON_TERMS, output, flags=re.IGNORECASE):
        violations.append("Use third-person only.")

    for term in URGENCY_TERMS:
        if re.search(term, output, flags=re.IGNORECASE):
            violations.append("Do not use urgency language.")
            break

    for term in BLAME_TERMS:
        if re.search(term, output, flags=re.IGNORECASE):
            violations.append("Do not assign blame.")
            break

    problem = _extract_section(output, "Problem Statement")
    solution = _extract_section(output, "Solution Statement") or _extract_section(output, "Proposed Solution")

    if not problem:
        violations.append("Problem Statement section must not be empty.")
    if not solution:
        violations.append("Solution Statement section must not be empty.")

    if problem:
        issue_parts = _extract_subsection(problem, "Issue Part Number(s)", PROBLEM_SUBSECTION_NAMES) or \
                      _extract_subsection(problem, "Issue Part Numbers", PROBLEM_SUBSECTION_NAMES)
        problem_desc = _extract_subsection(problem, "Problem Description", PROBLEM_SUBSECTION_NAMES)
        impact = _extract_subsection(problem, "Impact Details", PROBLEM_SUBSECTION_NAMES)

        if not issue_parts:
            violations.append("Problem Statement must include 'Issue Part Number(s):'.")
        if not problem_desc:
            violations.append("Problem Statement must include non-empty 'Problem Description:'.")
        if not impact:
            violations.append("Problem Statement must include non-empty 'Impact Details:'.")

        if problem_desc and not _has_bullet_lines(problem_desc):
            violations.append("Problem Description must use point-wise bullets with short complete sentences.")
        if impact and not _has_bullet_lines(impact):
            violations.append("Impact Details must use bullets.")

    if solution:
        solution_desc = _extract_subsection(solution, "Solution Description", SOLUTION_SUBSECTION_NAMES)
        if solution_desc and not _has_bullet_lines(solution_desc):
            violations.append("Solution Description must use point-wise bullets with short complete sentences.")

        if re.search(r"\bPart\s*Number\s*:", issue_parts or "", flags=re.IGNORECASE):
            violations.append("Issue Part Number(s) should not repeat 'Part Number:' labels.")

        for pat in DISALLOWED_PROBLEM_TERMS:
            if re.search(pat, "\n".join([problem_desc, impact]), flags=re.IGNORECASE):
                violations.append("Problem Statement contains solution language.")
                break

    source_parts = set(extract_part_numbers(source_text))
    out_parts = set(extract_part_numbers(output))
    if source_parts and not source_parts.issubset(out_parts):
        missing = sorted(source_parts - out_parts)
        violations.append(f"Missing part numbers from output: {', '.join(missing[:5])}")

    extra_parts = sorted(out_parts - source_parts)
    if extra_parts:
        violations.append(f"Output contains unknown part numbers: {', '.join(extra_parts[:5])}")

    return violations


def _is_supply_chain_shortage_scenario(text: str) -> bool:
    """Return True when the source text describes a supply shortage / excess-demand scenario.

    A single match of any _SC_STRONG_KEYWORDS item is sufficient.
    Otherwise, 2+ matches from _SC_SHORTAGE_KEYWORDS are required.
    """
    if not text:
        return False
    lower = text.lower()
    if any(kw in lower for kw in _SC_STRONG_KEYWORDS):
        return True
    matches = sum(1 for kw in _SC_SHORTAGE_KEYWORDS if kw in lower)
    return matches >= 2


def _sanitize_supply_chain_numbers(text: str) -> str:
    """Remove PO numbers and inline inventory/demand quantities from a statement.

    Post-processing safety net applied to LLM output in supply chain shortage scenarios.
    """
    if not text:
        return text
    text = _SC_PO_PATTERN.sub("", text)
    text = _SC_QTY_PATTERN.sub("", text)
    # Strip bare 10-digit sequences — AM PO numbers are always 10 digits.
    # Catches comma-separated lists of bare numbers (e.g. "4521626611, 4521643040, and 4521618014")
    text = re.sub(
        r"\b\d{10}\b(?:[,\s]+(?:and\s+)?\d{10}\b)*",
        "",
        text,
        flags=re.IGNORECASE,
    )
    # Strip residual PO action phrases (the approved-actions wording may survive as fragments)
    text = re.sub(r"\bopen\s+P\.?O\.?s?\b", "outstanding orders", text, flags=re.IGNORECASE)
    text = re.sub(r"\bopen\s+purchase\s+orders?\b", "outstanding orders", text, flags=re.IGNORECASE)
    # Clean up artifacts: trailing commas, orphaned conjunctions
    text = re.sub(r",[ \t]*,", ",", text)
    text = re.sub(r"[ \t]+,", ",", text)
    text = re.sub(r"\bare\s+affected\.?\s*", "", text, flags=re.IGNORECASE)
    # Collapse only horizontal whitespace (spaces/tabs) — NEVER collapse newlines.
    # Newlines separate bullet points and must be preserved.
    text = re.sub(r"[ \t]{2,}", " ", text)
    text = re.sub(r"^[ \t]*[,;][ \t]*", "", text, flags=re.MULTILINE)
    return text.strip()


def _scrub_sc_source_text(text: str) -> str:
    """Strip inventory quantities, demand figures, PO numbers, and date/quarter-coverage
    references from source_text BEFORE sending to the LLM in supply chain shortage scenarios.

    This is the primary enforcement layer: if the LLM never sees the specific numbers,
    it cannot reproduce them in the generated output.
    """
    if not text:
        return text
    # Remove PO number sequences
    text = _SC_PO_PATTERN.sub("", text)
    # Remove inline quantity references (numeric and text forms)
    text = _SC_QTY_PATTERN.sub("", text)
    # Remove orphaned "are affected" phrase left by PO removal
    text = re.sub(r"\bare\s+affected\.?\s*", "", text, flags=re.IGNORECASE)
    # Remove supply status timestamps: "as of 09 Apr 2025"
    text = re.sub(
        r"\bas\s+of\s+\d{1,2}\s+(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\w*\s+\d{4}\b",
        "", text, flags=re.IGNORECASE,
    )
    text = re.sub(r"\bas\s+of\s+\d{1,2}[/-]\d{1,2}[/-]\d{2,4}\b", "", text, flags=re.IGNORECASE)
    # Remove quarter coverage references: "covering demand only through 25Q3"
    text = re.sub(r"covering\s+demand\s+(?:only\s+)?through\s+\S+", "", text, flags=re.IGNORECASE)
    text = re.sub(r"demand\s+(?:only\s+)?through\s+\d*[Qq]\d+", "", text, flags=re.IGNORECASE)
    text = re.sub(r"no\s+coverage\s+(?:exists\s+)?for\s+\d*[Qq]\d+(?:\s+and\s+beyond)?", "", text, flags=re.IGNORECASE)
    text = re.sub(r"\bthrough\s+\d{2}[Qq]\d+\b", "", text, flags=re.IGNORECASE)
    text = re.sub(r"\bfor\s+\d{2}[Qq]\d+(?:\s+and\s+beyond)?\b", "", text, flags=re.IGNORECASE)
    # Clean up artifacts
    text = re.sub(r",\s*,", ",", text)
    text = re.sub(r"\s+,", ",", text)
    text = re.sub(r"\s{2,}", " ", text)
    text = re.sub(r"^\s*[,;\.]+\s*", "", text, flags=re.MULTILINE)
    return text.strip()

# ----------------------------
def _call_databricks(prompt_text: str) -> str:
    if not DATABRICKS_URL or not DATABRICKS_API_KEY:
        return "Error: Databricks URL or API Key not set"

    headers = {
        "Authorization": f"Bearer {DATABRICKS_API_KEY}",
        "Content-Type": "application/json",
    }
    payload = {"messages": [{"role": "user", "content": prompt_text.strip()}]}

    try:
        resp = requests.post(DATABRICKS_URL, headers=headers, json=payload, timeout=60)
    except requests.RequestException as e:
        return f"Error: Request failed - {e}"

    if resp.status_code != 200:
        return f"Error: {resp.status_code} - {resp.text}"

    try:
        data = resp.json()
        content = data["choices"][0]["message"]["content"]
        if not content:
            return "Error: LLM returned empty content"
        return content
    except Exception:
        return f"Error: Unexpected response format - {resp.text}"


# ----------------------------
# Input Feature Detection (drives conditional prompt injection)
# ----------------------------
def _detect_input_features(user_text: str) -> dict:
    """
    Scans user_text and returns a dict of boolean flags.
    Used by _build_engineering_prompt() to include only needed prompt sections.
    """
    t = user_text.lower()
    has_esw = bool(re.search(r"\besw\s*(?:#|number|num|no\.?)?\s*\d", t, re.IGNORECASE))
    has_oms = bool(re.search(r"\b0251-\d{5}\b", t))
    has_from_to = bool(re.search(r"\bfrom\s*:", t) or re.search(r"\bto\s*:", t))
    has_supply_chain = bool(
        re.search(r"\b(?:on[\-\s]hand|demand|purchase\s+order|p\.?o\.?\s*#|shortage|inventory|coverage)\b", t)
    )
    has_eol_obsolete = bool(
        re.search(r"\b(?:eol|end[\s\-]+of[\s\-]+life|obsolete(?:d)?|obsolescence|replacement\s+p/n)\b", t)
    )
    has_reference_records = bool(
        re.search(r"\b(?:sps|pcr|project|esw|qn|ecr|reference\s+records?)\b", t)
    )
    return {
        "has_esw": has_esw,
        "has_oms": has_oms,
        "has_from_to": has_from_to,
        "has_supply_chain": has_supply_chain,
        "has_eol_obsolete": has_eol_obsolete,
        "has_reference_records": has_reference_records,
    }

def reframe_problem(user_text: str, supply_chain_scenario: bool = False) -> str:
    prompt = _build_engineering_prompt(user_text, supply_chain_scenario)
    draft = _call_databricks(prompt)
    if draft.startswith("Error:"):
        return draft

    draft = clean_ai_output(draft)
    draft = remove_unwanted_sections(draft)
    draft = _remove_supplier_rev_tokens(draft)

    # Apply low-cost deterministic cleanup before any second pass.
    draft = re.sub(r"\bSPS\s*#?\s*\d+\b", "", draft, flags=re.IGNORECASE)
    if supply_chain_scenario:
        draft = _sanitize_supply_chain_numbers(draft)

    def _critical_violations(items: List[str]) -> List[str]:
        return [
            v for v in items
            if any(kw in v for kw in ("SPS", "Missing", "must not be empty"))
        ]

    violations = validate_engineering_output(draft, user_text)
    critical_violations = _critical_violations(violations)

    if not critical_violations:
        return draft

    # Lightweight deterministic normalization pass (no extra LLM call).
    title = _normalize_title_with_part_number(_extract_title(draft).strip(), user_text)
    problem_section = _extract_section(draft, "Problem Statement").strip()
    solution_section = _extract_section(draft, "Solution Statement").strip() or _extract_section(draft, "Proposed Solution").strip()

    normalized_problem = _normalize_problem_statement(problem_section, user_text)
    normalized_solution = _normalize_solution_statement(solution_section, user_text)
    deterministic = (
        f"Title:\n{title}\n\n"
        f"Problem Statement:\n{normalized_problem}\n\n"
        f"Solution Statement:\n{normalized_solution}"
    ).strip()
    deterministic = clean_ai_output(deterministic)
    deterministic = remove_unwanted_sections(deterministic)
    deterministic = _remove_supplier_rev_tokens(deterministic)
    deterministic = re.sub(r"\bSPS\s*#?\s*\d+\b", "", deterministic, flags=re.IGNORECASE)
    if supply_chain_scenario:
        deterministic = _sanitize_supply_chain_numbers(deterministic)

    deterministic_violations = validate_engineering_output(deterministic, user_text)
    deterministic_critical = _critical_violations(deterministic_violations)
    if len(deterministic_critical) < len(critical_violations):
        draft = deterministic
        critical_violations = deterministic_critical

    # Short corrective prompt fallback only when critical violations remain.
    if critical_violations:
        fix_prompt = f"""
Fix only these violations in the current draft:

- {"\n- ".join(critical_violations)}

Return the full corrected output using only these top-level headers:
Title:
Problem Statement:
Solution Statement:

Do not add explanations. Preserve existing valid content and edit minimally.

Current draft:
{draft}
""".strip()
        revised = _call_databricks(fix_prompt)
        if not revised.startswith("Error:"):
            revised = clean_ai_output(revised)
            revised = remove_unwanted_sections(revised)
            revised = _remove_supplier_rev_tokens(revised)
            revised = re.sub(r"\bSPS\s*#?\s*\d+\b", "", revised, flags=re.IGNORECASE)
            if supply_chain_scenario:
                revised = _sanitize_supply_chain_numbers(revised)

            revised_violations = validate_engineering_output(revised, user_text)
            revised_critical = _critical_violations(revised_violations)
            if len(revised_critical) <= len(critical_violations):
                draft = revised

    return draft

def _any_tab_checked(payload: Dict[str, Any]) -> bool:
    flags = payload.get("include_tabs_flags") or {}
    return any(bool(v) for v in flags.values())


# Section-header patterns that can appear inside raw ec_problem / ec_solution
# database fields. Stripping them prevents the LLM from generating nested
# section headers in its output, which would break the Title/Problem/Solution
# regex parsers in generate_full_pss.
_EC_FIELD_SECTION_HEADERS = re.compile(
    r"^\s*(?:Title|Problem\s+Statement|Solution\s+Statement|Proposed\s+Solution"
    r"|Problem\s+Description|Solution\s+Description|Reference\s+Change\s+Summary"
    r"|Affected(?:/Impacted)?\s+Part\s+Numbers?|Issue\s+Part\s+Numbers?"
    r"|Impact\s+Details|Benefits\s+of\s+the\s+Proposed\s+Solution)\s*:\s*$",
    re.IGNORECASE | re.MULTILINE,
)


def _sanitize_ec_field(text: str) -> str:
    """Remove embedded top-level section headers from an ECR problem/solution field.

    Some ECR records store already-formatted content (e.g. 'Problem Description:\\n...')
    inside ec_problem or ec_solution. Those embedded headers confuse the LLM into
    generating nested/duplicated section markers that break regex-based parsing.
    This function removes those header lines while preserving all other content.
    """
    if not text:
        return ""
    cleaned = _EC_FIELD_SECTION_HEADERS.sub("", text.strip())
    # Collapse any triple+ blank lines introduced by the removal.
    cleaned = re.sub(r"\n{3,}", "\n\n", cleaned)
    return cleaned.strip()


def _compose_user_text(payload: Dict[str, Any]) -> str:
    chunks: List[str] = []

    short_title = (payload.get("short_title") or "").strip()
    reason_code = (payload.get("reason_code") or "").strip()
    scope_parts = payload.get("scope_parts") or []
    refs = payload.get("reference_inputs") or {}
    current_problem = (payload.get("current_problem_text") or "").strip()
    current_solution = str(
        payload.get("current_solution_text") or payload.get("proposed_solution_text") or ""
    ).strip()
    solution_regen_requested = bool(payload.get("solution_regeneration_requested"))
    solution_regen_reason = str(payload.get("solution_regeneration_reason") or "").strip()
    selected_tab_payload = payload.get("selected_tab_payload") or {}

    # ------------------------------------------------------------------
    # PCR-sourced context (Question 1 – Databricks-enriched path)
    # ------------------------------------------------------------------
    pcr_records = payload.get("pcr_records") or []
    derived_psn = payload.get("derived_psn") or {}
    skipped_pcrs = payload.get("skipped_pcrs") or []
    project_records = payload.get("project_records") or []
    skipped_projects = payload.get("skipped_projects") or []
    sps_records = payload.get("sps_records") or []
    esw_records = payload.get("esw_records") or []
    reference_ecr_records = payload.get("reference_ecr_records") or []
    qn_records = payload.get("qn_records") or []

    if pcr_records:
        pcr_lines: List[str] = []
        for rec in pcr_records:
            pcr_id   = rec.get("pcr_id", "")
            status   = rec.get("status", "")
            problem  = _sanitize_ec_field(rec.get("problem") or "")
            solution = _sanitize_ec_field(rec.get("solution") or "")
            psn      = rec.get("psnnumber") or ""
            entry = (
                f"PCR {pcr_id} (Status: {status}):\n"
                f"  Problem: {problem}\n"
                f"  Solution: {solution}"
            )
            if psn:
                entry += f"\n  Associated PSN: {psn}"
            pcr_lines.append(entry)
        chunks.append("PCR Records from Databricks:\n\n" + "\n\n".join(pcr_lines))

    if derived_psn.get("answer") == "Yes":
        psn_nums = ", ".join(str(n) for n in (derived_psn.get("numbers") or []))
        chunks.append(f"PSN: Yes – PSN Number(s): {psn_nums}")
    elif derived_psn.get("answer") == "No":
        chunks.append("PSN: No – No associated PSN found in the PCR records.")

    if skipped_pcrs:
        skip_lines = [
            f"  PCR {s.get('pcr_id')} – Status: {s.get('status')}"
            for s in skipped_pcrs
        ]
        chunks.append(
            "Note – the following PCR(s) were skipped (inactive status) "
            "and are not included in the summary:\n" + "\n".join(skip_lines)
        )

    if project_records:
        project_lines: List[str] = []
        for rec in project_records:
            project_id = rec.get("project_id", "")
            status = rec.get("status", "")
            name = (rec.get("project_name") or "").strip()
            defined_scope = (rec.get("defined_scope") or "").strip()
            deliverables = (rec.get("deliverables") or "").strip()
            entry = (
                f"Project {project_id} (Status: {status}):\n"
                f"  Name: {name}\n"
                f"  Defined Scope (Problem Basis): {defined_scope}\n"
                f"  Deliverables (Solution Basis): {deliverables}"
            )
            project_lines.append(entry)
        chunks.append("Project Records from Databricks:\n\n" + "\n\n".join(project_lines))

    if skipped_projects:
        skip_project_lines = [
            f"  Project {s.get('project_id')} - Status: {s.get('status')}"
            for s in skipped_projects
        ]
        chunks.append(
            "Note - the following Project(s) were skipped (inactive status) "
            "and are not included in the summary:\n" + "\n".join(skip_project_lines)
        )

    def _append_ec_record_chunk(header: str, records: List[Dict[str, Any]], summary_label: str) -> None:
        if not records:
            return
        lines: List[str] = []
        for rec in records:
            ec_number = rec.get("ec_number") or rec.get("sps_id") or ""
            status = rec.get("status", "")
            title = (rec.get("title") or "").strip()
            problem = (rec.get("problem") or "").strip()
            solution = (rec.get("solution") or rec.get("proposed_solution") or "").strip()
            entry_lines = [f"{header} {ec_number} (Status: {status}):"]
            if title:
                entry_lines.append(f"  Title: {title}")
            entry_lines.append(f"  {summary_label}: {problem}")
            entry_lines.append(f"  Supporting Solution/Change: {solution}")
            lines.append("\n".join(entry_lines))
        chunks.append(f"{header} Records from Databricks:\n\n" + "\n\n".join(lines))

    def _append_sps_record_chunk(records: List[Dict[str, Any]]) -> None:
        """Like _append_ec_record_chunk but flags SPS as supplier-to-AM communication."""
        if not records:
            return
        lines: List[str] = []
        for rec in records:
            sps_id = rec.get("sps_id") or rec.get("ec_number") or ""
            status = rec.get("status", "")
            part_number = (rec.get("part_number") or "").strip()
            part_description = (rec.get("part_description") or "").strip()
            problem = (rec.get("problem") or "").strip()
            # Both proposed_solution (sps_problem.problem_soultion) and solution
            # (sps_solution.solution) may contain distinct, non-overlapping content.
            # Concatenate both so neither is silently dropped.
            proposed_sol = (rec.get("proposed_solution") or "").strip()
            extra_sol = (rec.get("solution") or "").strip()
            if proposed_sol and extra_sol and proposed_sol != extra_sol:
                combined_solution = proposed_sol + "\n  " + extra_sol
            elif proposed_sol:
                combined_solution = proposed_sol
            else:
                combined_solution = extra_sol
            entry = (
                f"SPS {sps_id} (Status: {status}) — SUPPLIER COMMUNICATION TO APPLIED MATERIALS:\n"
                f"  Supplier-reported issue: {problem}\n"
                f"  Supplier-proposed change: {combined_solution}"
            )
            if part_number:
                entry += f"\n  Part Number: {part_number}"
            if part_description:
                entry += f"\n  Part Description: {part_description}"
            lines.append(entry)
        chunks.append(
            "SPS Records from Databricks (raw supplier voice — translate all first-person to third-person; treat as supplier-to-AM communication):\n\n"
            + "\n\n".join(lines)
        )

    _append_sps_record_chunk(sps_records)
    _append_ec_record_chunk("ESW", esw_records, "Key Objective")
    _append_ec_record_chunk("Reference ECR", reference_ecr_records, "Key Objective")
    _append_ec_record_chunk("QN", qn_records, "Key Objective")
    # ------------------------------------------------------------------

    if short_title:
        chunks.append(f"Short Title Input: {short_title}")
    if reason_code:
        chunks.append(f"Reason Code: {reason_code}")

    if scope_parts:
        chunks.append("Scope Parts:\n" + "\n".join([f"- {str(x).strip()}" for x in scope_parts if str(x).strip()]))

    ref_lines = []
    for k, v in refs.items():
        ans = str((v or {}).get("answer", "")).strip()
        txt = str((v or {}).get("text", "")).strip()
        if ans or txt:
            ref_lines.append(f"{k} | Answer: {ans or 'N/A'} | Text: {txt}")
    if ref_lines:
        chunks.append("Reference Inputs:\n" + "\n".join(ref_lines))

    if current_problem:
        chunks.append("Problem Statement Input:\n" + current_problem)

    if current_solution:
        chunks.append("Proposed Solution Input:\n" + current_solution)

    if solution_regen_requested:
        chunks.append(
            "Solution Regeneration Request: Existing solution is incomplete or incorrect. "
            "Generate a complete replacement solution in senior-engineer style."
        )
        if solution_regen_reason:
            chunks.append("Solution Regeneration Note:\n" + solution_regen_reason)

    if selected_tab_payload:
        chunks.append("Selected Tab Payload:\n" + json.dumps(selected_tab_payload, ensure_ascii=False, indent=2))

    text = "\n\n".join(chunks).strip()

    bom = build_bom_context(text)
    replacement = build_replacement_context(text)

    if bom:
        text = f"{text}\n\n{bom}"
    if replacement:
        text = f"{text}\n\n{replacement}"

    # ---------------------------------------------------------------
    # Auto-BOM enrichment: if multiple part numbers are present,
    # query Where Used to resolve parent/child relationships so the
    # LLM can correctly classify Affected (parent) vs Issue (child).
    # ---------------------------------------------------------------
    wu_bom = _build_bom_context_from_where_used(text)
    if wu_bom:
        text = f"{text}\n\n{wu_bom}"

    return text


def _build_bom_context_from_where_used(source_text: str) -> str:
    """
    If two or more AM part numbers are found in *source_text*, query Databricks
    via ``where_used_query.fetch_where_used`` to discover the actual BOM
    parent-child relationships between them.

    Only pairs where BOTH the parent and child appear in the original input PN
    set are recorded — this prevents far-ancestor BOM noise from polluting the
    context.

    Returns a formatted string compatible with ``extract_bom_relationships()``
    so that ``_derive_part_sections_from_bom`` can automatically assign the
    correct parent to Issue Part Number(s) where applicable and child to Issue Part Number(s).

    Returns an empty string when:
    - fewer than two part numbers are present in source_text
    - ``where_used_query`` is not importable (graceful optional dependency)
    - the Databricks query fails for any reason (non-blocking fallback)
    - no parent-child pair is found among the input part numbers
    """
    pns = extract_part_numbers(source_text)
    if len(pns) < 2:
        return ""

    try:
        from where_used_query import fetch_where_used  # optional local dependency
    except ImportError:
        return ""

    pn_set = {p.upper() for p in pns}
    relationships: List[Tuple[str, str]] = []  # (parent, child)
    seen_pairs: set = set()

    try:
        # max_level=1 fetches only immediate parents — sufficient to detect
        # direct parent-child links and fast even for large BOMs.
        rows = fetch_where_used(list(pn_set), max_level=1)
    except Exception:
        return ""

    for row in rows:
        wu_level = str(row.get("wu_level", "")).strip()
        if wu_level != "1":
            continue
        # 'input_part' is the queried child PN; 'part' is the discovered parent.
        child = str(row.get("input_part") or "").strip().upper()
        parent = str(row.get("part") or "").strip().upper()
        if not child or not parent or child == parent:
            continue
        # Only capture pairs where both ends are present in the input PN set.
        if child in pn_set and parent in pn_set:
            pair = (parent, child)
            if pair not in seen_pairs:
                seen_pairs.add(pair)
                relationships.append(pair)

    if not relationships:
        return ""

    lines = [
        "BOM Relationships from Where Used query "
        "(Parent = used-in assembly; Child = component part):"
    ]
    for i, (parent, child) in enumerate(relationships, start=1):
        lines.append(
            f"{i}. Parent Part Number: {parent}; "
            f"Child Part Number: {child}; "
            f"Used in Part Number: {parent}"
        )
    return "\n".join(lines)


def _extract_sps_issue_part_descriptions(source_text: str) -> Dict[str, str]:
    """Extract Issue Part Number -> Description from SPS records only."""
    pn_desc: Dict[str, str] = {}
    if not source_text:
        return pn_desc

    in_sps_section = False
    current_pn = ""
    for raw in source_text.splitlines():
        line = raw.strip()
        if not line:
            continue

        if line.lower().startswith("sps records from databricks"):
            in_sps_section = True
            current_pn = ""
            continue

        # End SPS section at next clear top-level header.
        if in_sps_section and re.match(
            r"^(?:ESW|Reference\s+ECR|QN|Short\s+Title\s+Input|Reason\s+Code|Scope\s+Parts|Reference\s+Inputs|Problem\s+Statement\s+Input|Proposed\s+Solution\s+Input|Selected\s+Tab\s+Payload)\s*:",
            line,
            flags=re.IGNORECASE,
        ):
            break

        if not in_sps_section:
            continue

        m_pn = re.search(
            rf"\bPart\s*Number\s*:\s*({PN_FRAGMENT})\b",
            line,
            flags=re.IGNORECASE,
        )
        if m_pn:
            current_pn = m_pn.group(1).upper()
            if current_pn not in pn_desc:
                pn_desc[current_pn] = ""
            continue

        m_desc = re.search(r"\bPart\s*Description\s*:\s*(.+)$", line, flags=re.IGNORECASE)
        if m_desc and current_pn:
            desc = re.sub(r"\s+", " ", m_desc.group(1)).strip(" -:;,.")
            if desc:
                pn_desc[current_pn] = desc

    # Keep only entries that actually look like AM part numbers.
    return {pn: desc for pn, desc in pn_desc.items() if extract_part_numbers(pn)}


def _lookup_parent_parts_from_where_used(issue_parts: List[str], max_level: int = 1) -> List[str]:
    """Resolve immediate parent part numbers from Databricks where-used for issue parts."""
    children = _unique_preserve([p for p in issue_parts if p])
    if not children:
        return []

    try:
        from where_used_query import fetch_where_used  # optional local dependency
    except ImportError:
        return []

    try:
        rows = fetch_where_used(children, max_level=max_level)
    except Exception:
        return []

    parents: List[str] = []
    child_set = {c.upper() for c in children}
    for row in rows:
        if str(row.get("wu_level", "")).strip() != "1":
            continue
        child = str(row.get("input_part") or "").strip().upper()
        parent = str(row.get("part") or "").strip().upper()
        if not child or not parent or child == parent:
            continue
        if child not in child_set:
            continue
        if parent not in parents:
            parents.append(parent)
    return parents


def _rewrite_bullets_to_imperative(problem_desc: str, source_text: str = "") -> str:
    """
    Rewrites Problem Description bullet points from passive/negative phrasing
    into active imperative drawing-update statements, prefixed by a summary
    header.  Falls back to the original problem_desc on any LLM error.
    """
    if not problem_desc or not problem_desc.strip():
        return problem_desc

    # Detect primary part number from bullets first, then source context.
    pns = extract_part_numbers(problem_desc) or extract_part_numbers(source_text)
    primary_pn = pns[0] if pns else ""

    dwg_ref = f"Assy DWG for {primary_pn}" if primary_pn else "Assy DWG"

    prompt = (
        "You are an NPI/Manufacturing Engineer writing a formal Engineering Change Notice (ECN).\n\n"
        "TASK: Rewrite the bullet-point Problem Description below into action-oriented imperative "
        "statements suitable for a drawing update summary.\n\n"
        "RULES:\n"
        "1. One line per action. No bullet dashes, numbers, or list prefixes.\n"
        "2. Start each line with an imperative verb: Update, Specify, Define, Add note, "
        "Add handling note, Revise, Clarify, etc.\n"
        "3. Convert passive/negative sentences (\"X is not specified\") to active imperatives "
        "(\"Specify X as Y\").\n"
        "4. Keep all values, tolerances, part numbers, and technical specifics verbatim.\n"
        f"5. Do NOT include '{dwg_ref}' or any similar DWG reference in any output line — "
        "the header already identifies the drawing.\n"
        "6. Do NOT mention any person or human name in the output.\n"
        "7. If any text in the bullets is written in ALL CAPS (e.g. 'DO NOT BEND THE CORRUGATED TUBE'), "
        "preserve it exactly as-is — do not rephrase, lowercase, or reformat it.\n"
        "8. If the source context contains reference configurations or approval "
        "details relevant to a bullet, include them concisely on the same line.\n"
        "9. Output ONLY the imperative lines — no header, no preamble, no explanation, "
        "no blank lines between items.\n"
        "10. PRESERVE STRUCTURE: If the input already contains indented sub-bullet lines "
        "(lines starting with two or more spaces), preserve that grouped structure exactly "
        "as-is — do NOT flatten sub-bullets into separate top-level lines.\n"
        "11. Do NOT introduce new actions, new documents, or new bullets beyond what is "
        "present in the input bullets.\n"
        "12. CRITICAL — PART NUMBERS ONLY: After every part number, write the part number "
        "ONLY. Do NOT add any part name, part description, or parenthetical text "
        "immediately after the part number. "
        "WRONG: 'P/N 3310-00443 (2 INCH INDICATING PRESSURE TRANSMITTER)' "
        "RIGHT: 'P/N 3310-00443'.\n"
        "13. SENTENCE LENGTH: Each output line must be short, direct, and complete — prefer up to about 22 words. "
        "If an input bullet is too long or chains multiple actions with ' - ' dashes or "
        "semicolons, split it into separate output lines (one per action).\n\n"
        "PROBLEM DESCRIPTION BULLETS:\n"
        f"{problem_desc.strip()}\n\n"
        "SOURCE CONTEXT (for enrichment only):\n"
        f"{(source_text or '').strip()[:2000]}"
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return problem_desc  # graceful fallback

    # Strip any stray bullet/number prefixes from LLM output lines.
    # Also split any line that the LLM returned as a collapsed paragraph with
    # inline '. - Action' separators (Rule 13 compliance fallback).
    _SEP = re.compile(r"(?<=[a-zA-Z0-9.)]) - (?=[A-Z])", re.UNICODE)
    lines: List[str] = []
    for raw in result.strip().splitlines():
        ln = raw.strip()
        ln = re.sub(r"^[-*\u2022]\s*", "", ln)
        ln = re.sub(r"^\d+[.)]\s*", "", ln)
        ln = ln.strip()
        if not ln:
            continue
        # If this single LLM line contains chained actions, split it now.
        sub_parts = _SEP.split(ln)
        for part in sub_parts:
            part = part.strip()
            if part:
                lines.append(part)

    if not lines:
        return problem_desc  # fallback when LLM returned nothing useful

    return "\n".join(f"- {ln}" for ln in lines)


def _normalize_problem_statement(problem_section: str, source_text: str = "") -> str:
    if not problem_section:
        return ""

    issue_parts = _extract_subsection(problem_section, "Issue Part Number(s)", PROBLEM_SUBSECTION_NAMES)
    if not issue_parts:
        issue_parts = _extract_subsection(problem_section, "Issue Part Numbers", PROBLEM_SUBSECTION_NAMES)
    problem_desc = _extract_subsection(problem_section, "Problem Description", PROBLEM_SUBSECTION_NAMES)
    impact_details = _extract_subsection(problem_section, "Impact Details", PROBLEM_SUBSECTION_NAMES)

    problem_desc = _compress_repeated_part_number_labels(problem_desc)
    # Never carry Reason Code metadata into Problem Description narrative.
    problem_desc = re.sub(
        r"(?im)^\s*[-*•]?\s*Reason\s*Code\s*:\s*.*$",
        "",
        problem_desc,
    )
    problem_desc = re.sub(
        r"(?i)\bReason\s*Code\s*:\s*Beyond\s*Spec\s*Request\b[\s,;:.\-]*",
        "",
        problem_desc,
    )
    problem_desc = _remove_known_part_descriptions(problem_desc, source_text, remove_standalone=True)
    problem_desc = _break_long_sentences(problem_desc, max_length=220, max_words=22)
    problem_desc = _linebreak_after_fullstop(problem_desc)
    problem_desc = _inline_problem_description_from_to(problem_desc)
    problem_desc = _dedupe_problem_sentences(problem_desc, max_lines=0)
    problem_desc = _normalize_bullets(problem_desc)
    problem_desc = _dedupe_sps_in_bullets(problem_desc)
    problem_desc = _group_repeated_bullets(problem_desc)
    problem_desc = _reorganize_problem_bullets(problem_desc)
    problem_desc = _rewrite_problem_bullets_from_source(problem_desc, source_text)
    problem_desc = _limit_parent_reporting_bullets(problem_desc)
    problem_desc = _enforce_issue_parent_bullet_style(problem_desc, issue_parts, source_text)
    problem_desc = _dedupe_bullet_block(problem_desc)

    normalized_issue = _normalize_multiline_list(issue_parts) if issue_parts else ""

    # SPS-specific enforcement:
    # 1) Issue Part Number(s): always use SPS issue part number(s) with description.
    sps_issue_desc_map = _extract_sps_issue_part_descriptions(source_text)
    sps_issue_pns = _unique_preserve(list(sps_issue_desc_map.keys()))
    if sps_issue_pns:
        normalized_issue = "\n".join(
            f"{pn} - {sps_issue_desc_map[pn]}" if sps_issue_desc_map.get(pn) else pn
            for pn in sps_issue_pns
        )

    # Re-assert SPS-specific Issue section after any generic normalization.
    if sps_issue_pns:
        normalized_issue = "\n".join(
            f"{pn} - {sps_issue_desc_map[pn]}" if sps_issue_desc_map.get(pn) else pn
            for pn in sps_issue_pns
        )
    normalized_issue = _inject_single_part_description(normalized_issue, source_text)

    issue_pns = extract_part_numbers(normalized_issue)

    blocks: List[str] = []
    if problem_desc:
        blocks.append("Problem Description:\n" + problem_desc.strip())
    if impact_details:
        impact_details = _compress_repeated_part_number_labels(impact_details)
        blocks.append("Impact Details:\n" + _normalize_bullets(impact_details, crisp=True))
    blocks.append("Issue Part Number(s):\n" + (normalized_issue or "-"))

    return "\n\n".join(blocks).strip()


def _group_repeated_bullets(text: str) -> str:
    """
    Collapses bullet points that represent the same action applied to multiple documents.

    PREFIX is always tried first (handles sandwich and pure-prefix patterns).
    Pure SUFFIX detection is used only as fallback when common prefix < 5 words.

    SANDWICH pattern (long common prefix + varying doc ref + short common tail suffix):
      IN:  - Update 3D orientation for P/N X in OMS 0251-AAAA (Page 4), per SPS# 733703
           - Update 3D orientation for P/N X in OMS 0251-BBBB (Page 5), per SPS# 733703
      OUT: - Update 3D orientation for P/N X in the following OMS documents, per SPS# 733703:
             - OMS 0251-AAAA (Page 4)
             - OMS 0251-BBBB (Page 5)

    PREFIX+TRAILING-CLAUSE pattern (common prefix + varying doc ref + common tail clause):
      IN:  - Define flow direction for P/N X within OMS 0251-AAAA (Page 4) to eliminate ambiguity.
           - Define flow direction for P/N X within OMS 0251-BBBB (Page 5) to eliminate ambiguity.
      OUT: - Define flow direction for P/N X within the following OMS documents to eliminate ambiguity:
             - OMS 0251-AAAA (Page 4)
             - OMS 0251-BBBB (Page 5)

    PURE SUFFIX pattern (doc ref at start, action at end; common prefix < 5 words):
      IN:  - Update OMS 0251-AAAA (Page 4) to define flow direction for P/N X.
           - Update OMS 0251-BBBB (Page 5) to define flow direction for P/N X.
      OUT: - Update OMS documents to define flow direction for P/N X (applies to the following):
             - OMS 0251-AAAA (Page 4)
             - OMS 0251-BBBB (Page 5)
    """
    if not text:
        return text

    _DOC_KWORDS = {"OMS", "DWG", "ECR", "SOP", "BOM", "PCR", "WI", "SPS"}

    raw_lines = text.splitlines()
    top_level: List[Tuple[int, str]] = []
    sub_items_map: Dict[int, str] = {}

    for idx, raw in enumerate(raw_lines):
        if re.match(r"^\s{2,}", raw):          # already indented -> preserve as sub-item
            sub_items_map[idx] = raw
        else:
            stripped = re.sub(r"^[-*\u2022]\s*", "", raw.strip()).strip()
            top_level.append((idx, stripped))

    if len(top_level) < 2:
        return text

    def _words(s: str) -> List[str]:
        return s.split()

    def _cpfx(a: List[str], b: List[str]) -> int:
        """Number of matching leading words (case-insensitive)."""
        n = 0
        for x, y in zip(a, b):
            if x.lower() == y.lower():
                n += 1
            else:
                break
        return n

    def _csfx(a: List[str], b: List[str]) -> int:
        """Number of matching trailing words (case-insensitive, strip trailing punctuation)."""
        n = 0
        for x, y in zip(reversed(a), reversed(b)):
            if x.lower().rstrip(".,;:") == y.lower().rstrip(".,;:"):
                n += 1
            else:
                break
        return n

    def _min_tail_sfx(tails: List[List[str]]) -> int:
        """Minimum common trailing-word count across ALL tails."""
        if len(tails) < 2:
            return 0
        result = min(len(t) for t in tails)
        for t in tails[1:]:
            result = min(result, _csfx(tails[0], t))
        return result

    def _clean(s: str) -> str:
        return s.strip(" ,;:")

    n = len(top_level)
    used = [False] * n
    grouped: List[str] = []

    for i in range(n):
        if used[i]:
            continue
        orig_idx_i, text_i = top_level[i]
        words_i = _words(text_i)

        # ── PREFIX grouping — PRIMARY path (sandwich + pure-prefix) ─────────────
        best_pre_len = 0
        pre_group: List[int] = [i]
        for pre_try in range(min(30, len(words_i) - 1), 4, -1):
            candidate = [i]
            for j in range(i + 1, n):
                if used[j]:
                    continue
                words_j = _words(top_level[j][1])
                if _cpfx(words_i, words_j) >= pre_try:
                    candidate.append(j)
            if len(candidate) >= 2:
                best_pre_len = pre_try
                pre_group = candidate
                break

        if best_pre_len >= 5 and len(pre_group) >= 2:
            prefix_words = words_i[:best_pre_len]

            # Is the LAST word of the common prefix a document-type keyword?
            doc_type = ""
            last_pfx_word = prefix_words[-1].upper().strip("(),.:;")
            if last_pfx_word in _DOC_KWORDS:
                doc_type = last_pfx_word

            # Extract per-member tails (everything after the shared prefix)
            all_tails: List[List[str]] = [
                _words(top_level[k][1])[best_pre_len:] for k in pre_group
            ]

            # Find common tail SUFFIX across all tails (handles sandwich trailing part)
            raw_tail_suf = _min_tail_sfx(all_tails)
            # Safety: never consume the entire tail of any member
            tail_suf_len = raw_tail_suf
            for tw in all_tails:
                if tail_suf_len >= len(tw):
                    tail_suf_len = max(0, len(tw) - 1)

            # Build tail-suffix string (appended to header)
            tail_suffix_str = ""
            if tail_suf_len >= 1:
                tsw = list(all_tails[0][-tail_suf_len:])
                tsw[-1] = tsw[-1].rstrip(".")
                tail_suffix_str = _clean(" ".join(tsw))

            # Build sub-items: strip common tail suffix from each tail
            sub_lines: List[str] = []
            for k, tw in zip(pre_group, all_tails):
                core_words = tw[: len(tw) - tail_suf_len] if tail_suf_len else tw
                core_text = _clean(" ".join(core_words))
                # Prepend doc_type if the last prefix word was the doc keyword
                if doc_type and not core_text.upper().startswith(doc_type):
                    core_text = f"{doc_type} {core_text}"
                sub_lines.append(f"  - {core_text}")
                used[k] = True

            # Build header
            if doc_type:
                # Replace trailing doc_type keyword with "the following {doc_type} documents"
                hdr_pfx = " ".join(prefix_words[:-1]).rstrip(" ,;:")
                header = f"- {hdr_pfx} the following {doc_type} documents"
            else:
                hdr_pfx = " ".join(prefix_words).rstrip(" ,;:")
                header = f"- {hdr_pfx} in the following documents"

            if tail_suffix_str:
                # Space-only join when tail starts with a connector word (to, and, or…)
                _connectors = {"to", "and", "or", "which", "that", "where", "when", "as", "so"}
                first_tw = tail_suffix_str.split()[0].lower().rstrip(",;:") if tail_suffix_str else ""
                header += f" {tail_suffix_str}" if first_tw in _connectors else f", {tail_suffix_str}"

            header += ":"
            grouped.append(header)
            grouped.extend(sub_lines)
            continue

        # ── PURE SUFFIX grouping — FALLBACK (common prefix < 5 words) ───────────
        # Pattern: "Update OMS XXXX (Page N) to do Y" — doc ref at start, action at end.
        best_suf_len = 0
        suf_group: List[int] = [i]
        for suf_try in range(min(16, len(words_i) - 2), 4, -1):
            candidate = [i]
            for j in range(i + 1, n):
                if used[j]:
                    continue
                words_j = _words(top_level[j][1])
                if _csfx(words_i, words_j) >= suf_try:
                    candidate.append(j)
            if len(candidate) >= 2:
                best_suf_len = suf_try
                suf_group = candidate
                break

        if best_suf_len >= 5 and len(suf_group) >= 2:
            suffix_words = list(words_i[-best_suf_len:])
            suffix_words[-1] = suffix_words[-1].rstrip(".")
            suffix_str = " ".join(suffix_words)
            action_verb = words_i[0] if words_i else "Update"

            doc_type = ""
            for k in suf_group:
                vw = _words(top_level[k][1])[:-best_suf_len]
                for w in vw:
                    w_up = w.upper().strip("(),.:;")
                    if w_up in _DOC_KWORDS:
                        doc_type = w_up
                        break
                if doc_type:
                    break

            sub_lines = []
            for k in suf_group:
                vw = _words(top_level[k][1])[:-best_suf_len]
                if vw and vw[0].lower() == action_verb.lower():
                    vw = vw[1:]
                sub_text = _clean(" ".join(vw))
                if doc_type and not sub_text.upper().startswith(doc_type):
                    sub_text = f"{doc_type} {sub_text}"
                sub_lines.append(f"  - {sub_text}")
                used[k] = True

            # ── Build pivot-summary header ───────────────────────────────────────
            # Preferred: "Define X..., in the following OMS documents:"
            # If suffix starts with "to <verb>...", transform to imperative.
            _m_to = re.match(r"^to\s+(\S.*)$", suffix_str, flags=re.IGNORECASE)
            if _m_to and doc_type:
                action_phrase = _m_to.group(1).strip()
                action_phrase = action_phrase[0].upper() + action_phrase[1:]
                action_phrase = action_phrase.rstrip(".,;:")
                header = f"- {action_phrase}, in the following {doc_type} documents:"
            elif doc_type:
                clean_suf = suffix_str.rstrip(".,;:")
                header = f"- {action_verb} the following {doc_type} documents {clean_suf}:"
            else:
                header = f"- {action_verb} the following documents {suffix_str}:"
            grouped.append(header)
            grouped.extend(sub_lines)
            continue

        # ── No grouping — emit as-is ──────────────────────────────────────────
        orig_line = raw_lines[orig_idx_i]
        grouped.append(orig_line if orig_line.startswith("-") else f"- {text_i}")
        used[i] = True

    # Re-insert original sub-items (preserves already-grouped structure from input)
    for idx, sub_line in sub_items_map.items():
        grouped.append(sub_line)

    return "\n".join(grouped)


def _dedupe_sps_in_bullets(text: str) -> str:
    """
    Strips ALL SPS# references from any statement text (Problem or Solution).

    SPS numbers must NEVER appear in the output (per engineering policy).
    This post-processor removes any residual SPS# references the LLM may have
    emitted, handling all common forms:
      - "Per SPS# 733703, ..."     → leading prefix removed
      - "..., per SPS# 658420."   → trailing inline reference removed
      - "per SPS 733703"           → no-hash form removed

    Example (Solution):
      IN:  - Update OMS documents to define flow direction, per SPS# 733703:
      OUT: - Update OMS documents to define flow direction:

    Example (Problem):
      IN:  - Per SPS# 700817, the replacement P/N for 3310-00443 is not confirmed.
      OUT: - The replacement P/N for 3310-00443 is not confirmed.
    """
    if not text:
        return text

    # Leading "Per SPS# NNNNN, " or "Per SPS NNNNN, " at start of a bullet body
    SPS_LEADING = re.compile(
        r"^(\s*[-*\u2022]\s*)Per\s+SPS#?\s*\d+[,\s]+",
        re.IGNORECASE,
    )
    # Inline ", per SPS# NNNNN" or " per SPS NNNNN" anywhere in a line
    SPS_INLINE = re.compile(
        r",?\s*\bper\s+SPS#?\s*\d+\b",
        re.IGNORECASE,
    )

    out_lines: List[str] = []

    for line in text.splitlines():
        # Remove leading "Per SPS# NNN, " from bullet start
        line = SPS_LEADING.sub(lambda m: m.group(1), line)
        # Remove inline ", per SPS# NNN"
        if SPS_INLINE.search(line):
            line = SPS_INLINE.sub("", line)
            line = re.sub(r"[ \t]{2,}", " ", line).rstrip(" .,;:")
        if not line.strip().lstrip("-•* "):
            continue  # skip now-empty lines
        # Capitalise the first word of the bullet body if lowercased after strip
        m = re.match(r"^(\s*[-*\u2022]\s*)([a-z])", line)
        if m:
            line = line[:m.start(2)] + line[m.start(2)].upper() + line[m.end(2):]
        out_lines.append(line)

    return "\n".join(out_lines)


def _split_inline_dashes_to_bullets(text: str) -> str:
    """
    Splits top-level solution bullets that chain multiple actions using inline
    dash/em-dash separators into separate individual bullets.

    Handles all separator forms the LLM commonly emits:
      ' - '   regular hyphen with spaces        e.g. "...3310-00443 - Update BOM..."
      ' — '   em-dash with spaces               e.g. "...3310-00443. — Update BOM..."
      '.— '   period + em-dash (no space before) e.g. "...3310-00443.— Update BOM..."
      ' – '   en-dash with spaces               e.g. "...production – Communicate..."

    Example:
      IN:
        Evaluate replacement for 3310-00443. — Update BOM for 0090-09868. — Communicate to Supplier
      OUT:
        - Evaluate replacement for 3310-00443.
        - Update BOM for 0090-09868.
        - Communicate to Supplier

    Sub-bullets (indented 2+ spaces) are passed through unchanged so that
    OMS page-reference groups are not disturbed.
    Part numbers that contain hyphens (e.g. 3310-00443) are protected because they
    have no surrounding spaces.
    """
    if not text:
        return text

    # Matches sentence-level separators used by the LLM to chain multiple actions:
    # 1. Regular hyphen with spaces, preceded by a word char or period/paren:
    #    e.g. "...3170125. - Define..." or "...word - Update..."
    # 2. Em-dash (U+2014) optionally preceded by period/space: e.g. ".— " or ". — "
    # 3. En-dash (U+2013) with surrounding spaces: e.g. " – Update"
    # Right side must start with an uppercase letter (new action verb) so that
    # part-number hyphens (e.g. 3310-00443, digits on both sides) are never split.
    INLINE_SEP = re.compile(
        r"(?<=[a-zA-Z0-9.)]) - (?=[A-Z])"   # regular hyphen with spaces (after word/period/paren)
        r"|\.?\s*\u2014\s*(?=[A-Z])"          # em-dash, optional preceding period/space
        r"| \u2013 (?=[A-Z])",                # en-dash with spaces
        re.UNICODE,
    )

    out_lines: List[str] = []
    for line in text.splitlines():
        # Leave indented sub-bullets untouched
        if re.match(r"^\s{2,}", line):
            out_lines.append(line)
            continue

        # Detect bullet prefix to re-apply to each split piece
        prefix_m = re.match(r"^(\s*[-*\u2022]\s*)", line)
        prefix = prefix_m.group(1) if prefix_m else ""
        body = line[len(prefix):] if prefix else line

        parts = INLINE_SEP.split(body)
        if len(parts) > 1:
            for part in parts:
                part = part.strip().rstrip(".")
                if part:
                    out_lines.append((prefix if prefix else "- ") + part)
        else:
            out_lines.append(line)

    return "\n".join(out_lines)


def _reframe_long_solution_bullets(text: str, max_words: int = 15) -> str:
    """
    Uses the LLM to reframe any top-level solution bullet that exceeds max_words
    into a shorter, complete sentence of at most max_words words.

    Indented sub-bullets (OMS page references etc.) are passed through unchanged.
    If the LLM call fails, the original text is returned as a graceful fallback.
    """
    if not text:
        return text

    lines = text.splitlines()

    # Identify top-level bullets that exceed the word limit
    long_indices: List[int] = []
    for i, raw in enumerate(lines):
        if re.match(r"^\s{2,}", raw):
            continue  # skip indented sub-bullets
        body = re.sub(r"^[-*\u2022]\s*", "", raw.strip()).strip()
        if body and len(body.split()) > max_words:
            long_indices.append(i)

    if not long_indices:
        return text  # nothing to reframe

    # Build a numbered list of only the over-limit bullets for the LLM
    numbered: List[str] = []
    for seq, i in enumerate(long_indices, start=1):
        body = re.sub(r"^[-*\u2022]\s*", "", lines[i].strip()).strip()
        numbered.append(f"{seq}. {body}")

    prompt = (
        "You are an NPI/Manufacturing Engineer writing formal Engineering Change Notice bullets.\n\n"
        "TASK: Rewrite each numbered bullet below so it is at most "
        f"{max_words} words long while remaining a complete, clear, professional sentence.\n\n"
        "RULES:\n"
        "1. Keep the same action verb and technical meaning — do NOT omit key part numbers or actions.\n"
        "2. Use concise, professional language; standard abbreviations are applied automatically.\n"
        "3. Remove filler words (e.g. 'in order to', 'so as to', 'for the purpose of').\n"
        "4. Do NOT add new information or change the engineering intent.\n"
        "5. Do NOT write any SPS number or SPS# reference.\n"
        "6. Output ONLY the rewritten bullets, one per line, numbered to match the input "
        "(e.g. '1. <rewritten bullet>'). No extra explanation.\n\n"
        "BULLETS TO REWRITE:\n"
        + "\n".join(numbered)
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return text  # graceful fallback

    # Parse LLM response: expect "1. <text>", "2. <text>", …
    rewritten: Dict[int, str] = {}
    for raw_line in result.strip().splitlines():
        m = re.match(r"^(\d+)\.\s+(.+)$", raw_line.strip())
        if m:
            seq = int(m.group(1))
            rewritten[seq] = m.group(2).strip()

    # Apply rewrites back into the lines list
    for seq, i in enumerate(long_indices, start=1):
        if seq in rewritten and rewritten[seq]:
            prefix_m = re.match(r"^(\s*[-*\u2022]\s*)", lines[i])
            prefix = prefix_m.group(1) if prefix_m else "- "
            lines[i] = prefix + rewritten[seq]

    return "\n".join(lines)

def _complete_dangling_bullets(
    text: str,
    source_text: str = "",
    max_words: int = 15,
    imperative: bool = False,
) -> str:
    """
    Completes top-level bullets that end with dangling prepositions/fragments
    (for example ending in 'at', 'to', 'for').

    Uses a constrained LLM rewrite for only the affected bullets.
    """
    if not text:
        return text

    lines = text.splitlines()
    dangling_re = re.compile(
        r"\b(at|to|for|with|from|in|on|of|by|per|into|onto|than|as|via|within)\.?$",
        flags=re.IGNORECASE,
    )

    target_indices: List[int] = []
    for i, raw in enumerate(lines):
        if re.match(r"^\s{2,}", raw):
            continue
        body = re.sub(r"^[-*\u2022]\s*", "", raw.strip()).strip()
        if body and dangling_re.search(body):
            target_indices.append(i)

    if not target_indices:
        return text

    numbered: List[str] = []
    for seq, i in enumerate(target_indices, start=1):
        body = re.sub(r"^[-*\u2022]\s*", "", lines[i].strip()).strip()
        numbered.append(f"{seq}. {body}")

    style_line = (
        "Use imperative action style suitable for Solution Description."
        if imperative
        else "Use factual issue-only style suitable for Problem Description."
    )

    prompt = (
        "You are a senior Manufacturing Engineer finalizing ECN bullets.\n\n"
        "TASK: Complete each numbered bullet so it becomes a grammatically complete sentence "
        f"with at most {max_words} words.\n\n"
        "RULES:\n"
        "1. Keep original technical intent; do not add new assumptions.\n"
        "2. Keep existing part numbers and key terms unchanged.\n"
        "3. Do NOT write any SPS number or SPS# reference.\n"
        f"4. {style_line}\n"
        "5. Output ONLY numbered rewritten bullets matching input numbering.\n\n"
        "SOURCE CONTEXT (for wording disambiguation only):\n"
        f"{(source_text or '').strip()[:1200]}\n\n"
        "BULLETS TO COMPLETE:\n"
        + "\n".join(numbered)
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return text

    rewritten: Dict[int, str] = {}
    for raw_line in result.strip().splitlines():
        m = re.match(r"^(\d+)\.\s+(.+)$", raw_line.strip())
        if m:
            seq = int(m.group(1))
            rewritten[seq] = m.group(2).strip()

    for seq, i in enumerate(target_indices, start=1):
        if seq in rewritten and rewritten[seq]:
            prefix_m = re.match(r"^(\s*[-*\u2022]\s*)", lines[i])
            prefix = prefix_m.group(1) if prefix_m else "- "
            lines[i] = prefix + rewritten[seq]

    return "\n".join(lines)


def _reorganize_problem_bullets(text: str) -> str:
    """
    Uses the LLM to reorganize bullet points by identifying common themes/drivers
    and nesting related changes as sub-bullets under each theme heading.

    Applies to both Problem Description and Solution Description grouping scenarios.
    Preserves all technical details (P/Ns, revisions, suppliers).
    If the LLM call fails, the original text is returned as a graceful fallback.
    """
    if not text or not text.strip():
        return text

    lines = [l for l in text.splitlines() if l.strip()]
    if len(lines) < 5:
        return text  # not enough bullets to warrant reorganization

    prompt = (
        "You are an experienced Manufacturing Engineer writing a change control document.\n\n"
        "Analyze the following bullet list and:\n"
        "- Identify common themes or drivers (e.g., EOL, supplier change, firmware update, process improvement)\n"
        "- Create one high-level heading bullet per theme\n"
        "- Nest the related changes as sub-bullets under each heading\n\n"
        "Requirements:\n"
        "1. Use concise, professional, audit-ready language.\n"
        "2. Preserve all technical details (P/Ns, revisions, suppliers) exactly as given.\n"
        "3. Do NOT add new information, assumptions, or generic best-practice text.\n"
        "4. Ensure logical grouping (e.g., software changes, EOL-driven hardware changes, process-only changes).\n"
        "5. Separate software and hardware changes where applicable.\n"
        "6. Clearly indicate supplier EOL as the driver when present.\n"
        "7. Do NOT write any SPS number or SPS# reference.\n"
        "8. Output ONLY the reorganized bullet list — no preamble, no explanation.\n\n"
        "BULLET POINTS TO REORGANIZE:\n"
        + text
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return text  # graceful fallback
    return result.strip()


def _rewrite_problem_bullets_from_source(problem_desc: str, source_text: str = "") -> str:
    """Rewrite Problem Description bullets into concise, source-bound statements.

    CRITICAL: Every fact from the source must appear in the output.
    Use as many bullets as needed. Do not compress or drop content.
    """
    if not problem_desc or not problem_desc.strip():
        return problem_desc

    prompt = (
        "You are a senior manufacturing engineer rewriting a Problem Description for ECR documentation.\n\n"
        "TASK: Rewrite the problem bullets below into clear, easy-to-understand bullets.\n\n"
        "RULES:\n"
        "1. COMPLETE COVERAGE: Every factual issue from the source MUST appear as a bullet. Do NOT omit any fact.\n"
        "2. Use as many bullets as needed to cover the full source content. There is NO bullet count limit.\n"
        "3. Do NOT copy long prose verbatim. Reconstruct it into clear, simple bullets.\n"
        "4. Each bullet must be one complete, easy-to-understand sentence.\n"
        "5. Use only issue language: missing, incorrect, not documented, obsolete, inconsistent, changed, or affected.\n"
        "6. Do NOT introduce solution language, generic recommendations, or assumptions.\n"
        "7. Do NOT use SPS numbers. Do NOT repeat the same idea in multiple bullets.\n"
        "8. Preserve part numbers and supplier names exactly when present.\n"
        "9. When a supplier or manufacturer part number alone would be unclear, add a short description in brackets. Example: GA1068 (Gate Array), HCPL-0710-500ME (Photocoupler).\n"
        "10. For multi-change inputs covering software, hardware, connectors, PCB etc., organize bullets by category using short labels like Software Changes:, Hardware Changes:, Component Changes:.\n"
        "11. Output ONLY bullets, one per line, starting with '- '. No header or explanation.\n\n"
        "PROBLEM BULLETS TO REWRITE:\n"
        f"{problem_desc.strip()}\n\n"
        "SOURCE CONTEXT (for accuracy only — use it to verify completeness):\n"
        f"{(source_text or '').strip()[:4000]}"
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return problem_desc

    lines: List[str] = []
    for raw in result.strip().splitlines():
        ln = raw.strip()
        ln = re.sub(r"^[-*•]\s*", "", ln)
        if not ln:
            continue
        lines.append(f"- {ln}")

    if not lines:
        return problem_desc

    out = "\n".join(lines)
    out = _split_inline_dashes_to_bullets(out)
    return _dedupe_bullet_block(out)


def _limit_parent_reporting_bullets(text: str) -> str:
    """Keep parent-reporting context in at most one top-level bullet.

    When the LLM over-focuses on where-used/parent-chain narrative, this helper keeps
    only the first parent-reporting bullet and preserves the remaining issue-focused
    bullets in their original order.
    """
    if not text or not text.strip():
        return text

    lines = text.splitlines()
    top_items: List[str] = []
    for ln in lines:
        stripped = ln.strip()
        if not stripped:
            continue
        if re.match(r"^[-*\u2022]\s+", stripped):
            top_items.append(stripped)

    if len(top_items) < 2:
        return text

    parent_pat = re.compile(
        r"\b(parent|parents|where\s*used|reports?\s+to|used\s+in|bom\s+hierarchy|reporting\s+chain)\b",
        flags=re.IGNORECASE,
    )

    parent_bullets: List[str] = []
    issue_bullets: List[str] = []
    for bullet in top_items:
        body = re.sub(r"^[-*\u2022]\s+", "", bullet).strip()
        if parent_pat.search(body):
            parent_bullets.append(bullet)
        else:
            issue_bullets.append(bullet)

    if len(parent_bullets) <= 1:
        return text

    # Keep all parent bullets that mention DIFFERENT part numbers.
    seen_parent_pns: set = set()
    unique_parent_bullets: List[str] = []
    for pb in parent_bullets:
        pns = set(extract_part_numbers(pb))
        if pns and pns <= seen_parent_pns:
            continue  # same parts already covered
        seen_parent_pns |= pns
        unique_parent_bullets.append(pb)
    if not unique_parent_bullets:
        unique_parent_bullets = [parent_bullets[0]]

    rebuilt: List[str] = list(unique_parent_bullets)
    rebuilt.extend(issue_bullets)

    return "\n".join(rebuilt)


def _extract_supplier_mpn_for_part(source_text: str, am_pn: str) -> str:
    """Extract supplier MPN tied to an Applied part number from source context.

    Supports patterns like:
      AMAT PN 1270-03765 (BI2U-...)
      1270-03765(BI2U-...)
      P/N 1270-03765, Supplier P/N BI2U-...
    """
    if not source_text or not am_pn:
        return ""

    def _is_am_part(token: str) -> bool:
        return bool(re.fullmatch(PN_FRAGMENT, token or "", flags=re.IGNORECASE))

    # 1) Bracket immediately after Applied PN.
    m = re.search(
        rf"\b{re.escape(am_pn)}\b\s*\(([^)]+)\)",
        source_text,
        flags=re.IGNORECASE,
    )
    if m:
        cand = re.sub(r"\s+", " ", (m.group(1) or "").strip()).upper()
        if cand and not _is_am_part(cand):
            return cand

    # 2) Same line search with explicit supplier/manufacturer markers.
    am_up = am_pn.upper()
    for ln in (source_text or "").splitlines():
        ln_up = ln.upper()
        if am_up not in ln_up:
            continue
        m2 = re.search(
            r"(?:Supplier|Manufacturer|OEM)\s*P/?N\s*[:#]?\s*([A-Z0-9][A-Z0-9\-\/.]{4,})",
            ln,
            flags=re.IGNORECASE,
        )
        if m2:
            cand = (m2.group(1) or "").strip().upper()
            if cand and not _is_am_part(cand):
                return cand

    # 3) Fallback: supplier P/N appears before AMAT/Applied P/N reference.
    pat = re.compile(
        rf"\bP/?N\s*[:#]?\s*([A-Z0-9][A-Z0-9\-\/.]{{4,}}).*?"
        rf"(?:AMAT|Applied\s+Materials?)\s*(?:P/?N|Part\s*Number)\s*[:#]?\s*{re.escape(am_pn)}\b",
        flags=re.IGNORECASE | re.DOTALL,
    )
    m3 = pat.search(source_text)
    if m3:
        cand = (m3.group(1) or "").strip().upper()
        if cand and not _is_am_part(cand):
            return cand

    return ""


def _extract_supplier_mpn_for_issue(source_text: str, issue_pn: str) -> str:
    return _extract_supplier_mpn_for_part(source_text, issue_pn)


def _commodity_class_label(part_number: str) -> str:
    """Return a compact commodity class label for narrative use."""
    if not part_number:
        return ""
    m = re.search(r"(\d{4})", str(part_number))
    if not m:
        return ""
    code = int(m.group(1))
    if code == 15:
        return "Modified OEM"
    if code == 195:
        return "Spec Controlled OEM"
    if code >= 500:
        return "OEM"
    return ""


def _extract_supplier_name(source_text: str) -> str:
    """Extract supplier/manufacturer name from source text when available."""
    if not source_text:
        return "Supplier"

    patterns = [
        r"\bManufacturer\s+([A-Za-z][A-Za-z0-9&\- ]{1,40})\s+has\s+(?:discontinued|declared|obsoleted)",
        r"\bSupplier\s+([A-Za-z][A-Za-z0-9&\- ]{1,40})\s+has\s+(?:discontinued|declared|obsoleted)",
        r"\b([A-Za-z][A-Za-z0-9&\- ]{1,40})\s+has\s+(?:discontinued|declared|obsoleted)",
    ]
    for pat in patterns:
        m = re.search(pat, source_text, flags=re.IGNORECASE)
        if m:
            name = re.sub(r"\s+", " ", (m.group(1) or "").strip())
            name = re.sub(r"\b(AMAT|Applied\s+Materials)\b", "", name, flags=re.IGNORECASE).strip(" -")
            if name and len(name) <= 40:
                return name
    return "Supplier"


def _enforce_issue_parent_bullet_style(problem_desc: str, issue_text: str, source_text: str) -> str:
    """Apply SPS issue-focused Problem Description style constraints.

    Rules enforced:
    - Issue obsolete statements must refer to issue P/N (not parent P/N).
    - Applied P/N appears first; supplier MPN is shown in brackets when available.
    - Parent structuring is mentioned in only one bullet and placed as bullet 2.
    - Remove second-level parent chain phrases (e.g., "which is child of ...").
    """
    if not problem_desc or not problem_desc.strip():
        return problem_desc

    issue_pns = extract_part_numbers(issue_text or "")
    issue_pn = issue_pns[0] if issue_pns else ""
    supplier_mpn = _extract_supplier_mpn_for_issue(source_text, issue_pn) if issue_pn else ""
    supplier_name = _extract_supplier_name(source_text)
    commodity_label = _commodity_class_label(issue_pn) if issue_pn else ""
    bom_rels = extract_bom_relationships(source_text or "")
    first_level_parent = ""
    if issue_pn:
        for parent, child in bom_rels:
            if child.upper() == issue_pn.upper():
                first_level_parent = parent
                break

    bullets: List[str] = []
    for ln in problem_desc.splitlines():
        m = re.match(r"^\s*[-*\u2022]\s+(.+)$", ln.strip())
        if m:
            bullets.append(m.group(1).strip())

    if not bullets:
        return problem_desc

    obsolete_pat = re.compile(r"\b(obsolete|obsoleted|eol|end\s+of\s+life|discontinued?)\b", flags=re.IGNORECASE)
    parent_pat = re.compile(r"\b(parent|parents|where\s*used|reports?\s+to|used\s+in|assy|assembly|bom\s+hierarchy|child\s+component\s+of)\b", flags=re.IGNORECASE)
    second_level_pat = re.compile(
        rf",?\s*(?:which|that)\s+(?:is\s+)?(?:also\s+)?(?:a\s+)?child\s+of\s+\b{PN_FRAGMENT}\b",
        flags=re.IGNORECASE,
    )

    cleaned: List[str] = []
    for b in bullets:
        x = second_level_pat.sub("", b)
        x = re.sub(r"\s{2,}", " ", x).strip(" ,;:-")

        # Keep replacement-identification details out of Problem Description (solution content).
        if re.search(r"\b(identified\s+replacement|replacement\s+part\s+number|will\s+be\s+replaced\s+by|replace[d]?\s+by)\b", x, flags=re.IGNORECASE):
            continue

        if issue_pn and obsolete_pat.search(x):
            # Force supplier-MPN obsolescence style with commodity + parent in same bullet.
            prefix = f"Applied {commodity_label} P/N" if commodity_label else "Applied P/N"
            if supplier_mpn and first_level_parent:
                x = (
                    f"{supplier_name} declared Supplier P/N {supplier_mpn} obsolete for {prefix} {issue_pn}, "
                    f"which is a child component of Assy. {first_level_parent}."
                )
            elif supplier_mpn:
                x = f"{supplier_name} declared Supplier P/N {supplier_mpn} obsolete for {prefix} {issue_pn}."
            elif first_level_parent:
                x = f"{supplier_name} reported OEM obsolescence impacting {prefix} {issue_pn}, a child of Assy. {first_level_parent}."
            else:
                x = f"{supplier_name} reported OEM obsolescence impacting {prefix} {issue_pn}."

        if issue_pn and first_level_parent and parent_pat.search(x) and not obsolete_pat.search(x):
            if supplier_mpn:
                x = (
                    f"Applied P/N {issue_pn} (Supplier P/N {supplier_mpn}) "
                    f"is a child component of Assy. {first_level_parent}"
                )
            else:
                x = f"Applied P/N {issue_pn} is a child component of Assy. {first_level_parent}"

        if x and not x.endswith((".", "!", "?")):
            x += "."
        if x:
            cleaned.append(x)

    if not cleaned:
        return problem_desc

    # Keep only one parent-structure bullet.
    parent_idx = [i for i, b in enumerate(cleaned) if parent_pat.search(b)]
    if len(parent_idx) > 1:
        keep = parent_idx[0]
        cleaned = [b for i, b in enumerate(cleaned) if (i == keep or i not in parent_idx)]
        parent_idx = [i for i, b in enumerate(cleaned) if parent_pat.search(b)]

    # Place parent-structure bullet as 2nd bullet when possible.
    if parent_idx and len(cleaned) > 1:
        idx = parent_idx[0]
        pb = cleaned.pop(idx)
        cleaned.insert(1, pb)

    # Ensure supplier name is always present in Problem Description.
    supplier_present = any(re.search(rf"\b{re.escape(supplier_name)}\b", b, flags=re.IGNORECASE) for b in cleaned)
    if not supplier_present:
        cleaned.insert(0, f"{supplier_name} reported the issue for Applied P/N {issue_pn}." if issue_pn else f"{supplier_name} reported the issue.")

    return _dedupe_bullet_block("\n".join(f"- {b}" for b in cleaned))


def _normalize_solution_statement(solution_section: str, source_text: str = "") -> str:
    if not solution_section:
        return ""

    proposed_change = _extract_subsection(solution_section, "Proposed Change", SOLUTION_SUBSECTION_NAMES)
    solution_desc = _extract_subsection(solution_section, "Solution Description", SOLUTION_SUBSECTION_NAMES)
    benefits = _extract_subsection(solution_section, "Benefits of the Proposed Solution", SOLUTION_SUBSECTION_NAMES)

    proposed_change = _compress_repeated_part_number_labels(proposed_change)
    proposed_change = _format_from_to_changes(proposed_change)
    solution_desc = _compress_repeated_part_number_labels(solution_desc)
    solution_desc = _remove_known_part_descriptions(solution_desc, solution_section, remove_standalone=True)
    # NOTE: _break_long_sentences and _linebreak_after_fullstop are intentionally
    # skipped here — they collapse indented sub-bullet groups (OMS page references)
    # into a flat single line, destroying the grouped structure.
    solution_desc = _format_from_to_changes(solution_desc)
    solution_desc = _normalize_bullets(solution_desc)
    # Split any ' - ' chained actions into separate bullets before further processing.
    solution_desc = _split_inline_dashes_to_bullets(solution_desc)
    benefits = _compress_repeated_part_number_labels(benefits)

    # Backward-compatible parsing: if model still emits "Proposed Change",
    # fold it into a single "Solution Description" block.
    merged_solution_desc = "\n".join(
        [x.strip() for x in [proposed_change, solution_desc] if x and x.strip()]
    ).strip()
    merged_solution_desc = _normalize_bullets(merged_solution_desc)
    merged_solution_desc = _split_inline_dashes_to_bullets(merged_solution_desc)
    merged_solution_desc = _rewrite_bullets_to_imperative(merged_solution_desc, source_text)
    # Re-apply split after LLM rewrite in case the model collapsed bullets back to one line.
    merged_solution_desc = _split_inline_dashes_to_bullets(merged_solution_desc)
    # Strip any part descriptions the LLM may have re-introduced after rewriting.
    merged_solution_desc = _remove_known_part_descriptions(merged_solution_desc, source_text, remove_standalone=False)
    merged_solution_desc = _group_repeated_bullets(merged_solution_desc)
    merged_solution_desc = _dedupe_sps_in_bullets(merged_solution_desc)
    # Final safety-net: ensure no chained bullets remain in the assembled output.
    merged_solution_desc = _split_inline_dashes_to_bullets(merged_solution_desc)
    # LLM reframe: shorten any top-level bullet that still becomes too long.
    merged_solution_desc = _reframe_long_solution_bullets(merged_solution_desc, max_words=22)
    # Repair incomplete/truncated bullets ending with dangling prepositions.
    merged_solution_desc = _complete_dangling_bullets(merged_solution_desc, source_text, max_words=22, imperative=True)
    merged_solution_desc = _dedupe_bullet_block(merged_solution_desc)

    blocks: List[str] = []
    if merged_solution_desc:
        blocks.append("Solution Description:\n" + merged_solution_desc)
    if benefits:
        blocks.append("Benefits of the Proposed Solution:\n" + _normalize_bullets(benefits, crisp=True))

    return "\n\n".join(blocks).strip()


def _source_has_explicit_solution(payload: Dict[str, Any]) -> bool:
    """True if any source channel provides a concrete solution/proposed-change field."""
    if str(payload.get("current_solution_text") or "").strip():
        return True
    if str(payload.get("proposed_solution_text") or "").strip():
        return True

    record_groups = [
        payload.get("pcr_records") or [],
        payload.get("project_records") or [],
        payload.get("sps_records") or [],
        payload.get("esw_records") or [],
        payload.get("reference_ecr_records") or [],
        payload.get("qn_records") or [],
    ]

    for records in record_groups:
        for rec in records:
            if str(rec.get("solution") or "").strip():
                return True
            if str(rec.get("proposed_solution") or "").strip():
                return True
            if str(rec.get("deliverables") or "").strip():
                return True
            if str(rec.get("immediate_corrective_action") or "").strip():
                return True
    return False


def _normalize_source_block_text(text: str) -> str:
    if not text:
        return ""
    lines = [ln.rstrip() for ln in str(text).splitlines()]
    # Preserve original line-level content while collapsing excessive blank groups.
    joined = "\n".join(lines).strip()
    joined = re.sub(r"\n{3,}", "\n\n", joined)
    return joined

def _infer_solution_input_files(payload: Dict[str, Any]) -> List[str]:
    labels: List[str] = []
    if payload.get("sps_records"):
        labels.append("SPS")
    if payload.get("pcr_records"):
        labels.append("PCR")
    if payload.get("project_records"):
        labels.append("Project")
    if payload.get("esw_records"):
        labels.append("ESW")
    if payload.get("qn_records"):
        labels.append("QN")

    if not labels:
        labels = ["SPS", "PCR", "Project", "ESW", "QN"]
    return labels


def _extract_alternate_equivalent_parts(solution_text: str, primary_replacement: str) -> List[str]:
    if not solution_text:
        return []
    alternates: List[str] = []
    for raw in str(solution_text).splitlines():
        line = raw.strip()
        if not line:
            continue
        if not re.search(r"\b(alternate|alternative|equivalent|equiv\.?|substitute|interchangeable|drop[-\s]?in)\b", line, flags=re.IGNORECASE):
            continue
        for pn in extract_part_numbers(line):
            if pn != primary_replacement and pn not in alternates:
                alternates.append(pn)
    return alternates


def _extract_supporting_reasoning(solution_text: str) -> List[str]:
    if not solution_text:
        return []
    reasons: List[str] = []
    for raw in str(solution_text).splitlines():
        line = raw.strip()
        if not line:
            continue
        if re.search(r"\b(due to|because|to address|to avoid|to mitigate|to ensure|pending|therefore|hence|reason|rationale|eol|obsolete)\b", line, flags=re.IGNORECASE):
            if line not in reasons:
                reasons.append(line)
    return reasons


def _build_complete_problem_description(payload: Dict[str, Any]) -> str:
    entries = _collect_complete_problem_entries(payload)
    if not entries:
        return ""

    merged = "\n".join(text for _, text in entries if text).strip()
    if not merged:
        return ""
    cleaned = _normalize_bullets(merged)
    cleaned = _dedupe_bullet_block(cleaned)
    if not cleaned:
        return ""
    return "Problem Description:\n" + cleaned


def _replace_problem_description(problem_statement: str, new_problem_desc: str) -> str:
    if not new_problem_desc:
        return problem_statement

    if not problem_statement:
        return "Problem Description:\n" + new_problem_desc

    pattern = re.compile(
        r"(^\s*Problem\s+Description\s*:\s*)(.*?)(?=^\s*Impact\s+Details\s*:|\Z)",
        flags=re.IGNORECASE | re.MULTILINE | re.DOTALL,
    )
    if pattern.search(problem_statement):
        return pattern.sub(lambda m: m.group(1) + new_problem_desc + "\n", problem_statement)

    return problem_statement.strip() + "\n\nProblem Description:\n" + new_problem_desc


def _build_complete_solution_statement(payload: Dict[str, Any], existing_solution: str = "") -> str:
    entries = _collect_complete_solution_entries(payload)
    if not entries:
        file_labels = ", ".join(_infer_solution_input_files(payload))
        missing_msg = (
            "- No solution found in input files. "
            f"Please update solution in the input file(s): {file_labels}."
        )
        return "Solution Description:\n" + missing_msg

    source_solution_text = "\n".join(text for _, text in entries if text).strip()
    source_solution_desc = _normalize_bullets(source_solution_text)
    source_solution_desc = _split_inline_dashes_to_bullets(source_solution_desc)
    source_solution_desc = _rewrite_bullets_to_imperative(source_solution_desc, source_solution_text)
    source_solution_desc = _split_inline_dashes_to_bullets(source_solution_desc)
    source_solution_desc = _dedupe_bullet_block(source_solution_desc)

    existing_solution_desc = _extract_subsection(existing_solution, "Solution Description", SOLUTION_SUBSECTION_NAMES)
    if not existing_solution_desc:
        existing_solution_desc = existing_solution.strip()
    merged_solution_desc = _normalize_bullets(existing_solution_desc)
    merged_solution_desc = _split_inline_dashes_to_bullets(merged_solution_desc)
    merged_solution_desc = _dedupe_bullet_block(merged_solution_desc)

    # Keep LLM-reframed output as primary when available, but backfill from source if needed.
    final_solution_desc = merged_solution_desc or source_solution_desc

    # Preserve explicit source replacement mappings so critical part swaps are never dropped.
    replacement_pairs = extract_replacement_pairs(source_solution_text)
    for obsolete_pn, replacement_pn in replacement_pairs:
        if not obsolete_pn or not replacement_pn:
            continue
        if re.search(rf"\b{re.escape(obsolete_pn)}\b", final_solution_desc, flags=re.IGNORECASE) and re.search(
            rf"\b{re.escape(replacement_pn)}\b",
            final_solution_desc,
            flags=re.IGNORECASE,
        ):
            continue
        replacement_line = f"- Replace {obsolete_pn} with {replacement_pn}."
        final_solution_desc = (replacement_line + "\n" + final_solution_desc).strip() if final_solution_desc else replacement_line

    final_solution_desc = _dedupe_bullet_block(final_solution_desc)

    blocks: List[str] = []
    if final_solution_desc:
        blocks.append("Solution Description:\n" + final_solution_desc)

    benefits = _extract_subsection(existing_solution, "Benefits of the Proposed Solution", SOLUTION_SUBSECTION_NAMES)
    if benefits:
        blocks.append("Benefits of the Proposed Solution:\n" + _normalize_bullets(benefits, crisp=True))

    return "\n\n".join(blocks).strip()


def _looks_like_solution_bullet(line: str) -> bool:
    s = re.sub(r"^[-*•]\s*", "", str(line or "").strip())
    low = s.lower()
    if not s:
        return False

    # Hard-keep explicit issue/non-conformance statements in Problem Description.
    problem_markers = [
        "not specified", "not stated", "not documented", "does not",
        "is incorrect", "incorrect", "missing", "unavailable",
        "material shortage", "risk", "is not", "are not", "cannot",
        "no alternative", "no currently", "shortage",
    ]
    # Only keep as problem if the WHOLE line is issue-focused (no solution clause after semicolon).
    if any(m in low for m in problem_markers):
        # If a semicolon splits a problem clause from a solution clause, this
        # whole-line check will fail — caller should split on semicolon first.
        return False

    solution_markers = [
        "update", "specify", "define", "add note", "add handling",
        "please update", "please state", "please define",
        "communicated", "communicated to",
        "buyoff", "buyoff process", "established that",
        "confirmed", "confirmed during", "approved",
        "must be", "should be", "set to",
        "value of", "was communicated",
        "was established", "is required to be",
        "oriented", "orientation is",
        "routing condition", "must remain", "must be routed",
        "must be maintained", "must not be bent",
        "oms 0251-",  # OMS document update bullets belong in Solution
    ]
    return any(m in low for m in solution_markers)


def _split_mixed_bullet(line: str) -> Tuple[str, str]:
    """
    If a bullet contains a semicolon that separates a problem clause from a
    solution clause, split it and return (problem_part, solution_part).
    If no split is needed, returns (line, "").
    """
    s = re.sub(r"^[-*•]\s*", "", str(line or "").strip())
    if ";" not in s:
        return line, ""

    parts = [p.strip() for p in s.split(";")]
    problem_parts: List[str] = []
    solution_parts: List[str] = []
    for part in parts:
        if _looks_like_solution_bullet(part):
            solution_parts.append(part)
        else:
            problem_parts.append(part)

    if not solution_parts:
        return line, ""
    if not problem_parts:
        return "", line

    return ("- " + "; ".join(problem_parts).strip(".")+"."), \
           ("- " + "; ".join(solution_parts).strip(".")+".")


def _rebalance_problem_solution(
    normalized_problem: str,
    normalized_solution: str,
    source_text: str,
) -> Tuple[str, str]:
    """
    Always run: keep Problem Description strictly issue-only.
    Move solution-like bullets from Problem Description into Solution Description.
    Also splits mixed semicolon bullets that combine problem + solution in one line.
    """
    if not normalized_problem.strip():
        return normalized_problem, normalized_solution

    problem_desc = _extract_subsection(normalized_problem, "Problem Description", PROBLEM_SUBSECTION_NAMES)
    if not problem_desc.strip():
        return normalized_problem, normalized_solution

    original_lines = [ln.strip() for ln in problem_desc.splitlines() if ln.strip()]
    moved: List[str] = []
    kept: List[str] = []
    for ln in original_lines:
        # First try splitting a mixed bullet on semicolon.
        prob_part, sol_part = _split_mixed_bullet(ln)
        if sol_part:
            if prob_part:
                kept.append(prob_part)
            moved.append(sol_part)
        elif _looks_like_solution_bullet(ln):
            moved.append(ln)
        else:
            kept.append(ln)

    if not moved:
        return normalized_problem, normalized_solution

    # Rebuild Problem Statement blocks while preserving existing subsection structure.
    # Handle both canonical and ID-suffixed headers:
    #   Change Drivers:
    #   Change Drivers (PCR# ..., Project# ...):
    #   ESW Turn Around:
    #   ESW Turn Around (ESW# ...):
    change_drivers_header = "Change Drivers:"
    m_ref_header = re.search(
        r"^\s*(Change\s+Drivers(?:\s*\([^\n)]*\))?)\s*:",
        normalized_problem,
        flags=re.IGNORECASE | re.MULTILINE,
    )
    if m_ref_header:
        change_drivers_header = m_ref_header.group(1).strip() + ":"

    esw_turnaround_header = "ESW Turn Around:"
    m_esw_header = re.search(
        r"^\s*(ESW\s+Turn\s+Around(?:\s*\([^\n)]*\))?)\s*:",
        normalized_problem,
        flags=re.IGNORECASE | re.MULTILINE,
    )
    if m_esw_header:
        esw_turnaround_header = m_esw_header.group(1).strip() + ":"

    ref = _extract_subsection(normalized_problem, "Change Drivers", PROBLEM_SUBSECTION_NAMES)
    if not ref:
        m_ref = re.search(
            r"^\s*Change\s+Drivers(?:\s*\([^\n)]*\))?\s*:\s*(.*?)(?=^\s*(?:Affected/Impacted\s+Part\s+Numbers|Affected\s+Part\s+Number\(s\)|Affected\s+Part\s+Numbers|Issue\s+Part\s+Number\(s\)|Issue\s+Part\s+Numbers|Problem\s+Description|Impact\s+Details|ESW\s+Turn\s+Around(?:\s*\([^\n)]*\))?)\s*:|\Z)",
            normalized_problem,
            flags=re.IGNORECASE | re.MULTILINE | re.DOTALL,
        )
        ref = m_ref.group(1).strip() if m_ref else ""

    esw_turnaround = _extract_subsection(normalized_problem, "ESW Turn Around", PROBLEM_SUBSECTION_NAMES)
    if not esw_turnaround:
        m_esw = re.search(
            r"^\s*ESW\s+Turn\s+Around(?:\s*\([^\n)]*\))?\s*:\s*(.*?)(?=^\s*(?:Affected/Impacted\s+Part\s+Numbers|Affected\s+Part\s+Number\(s\)|Affected\s+Part\s+Numbers|Issue\s+Part\s+Number\(s\)|Issue\s+Part\s+Numbers|Problem\s+Description|Impact\s+Details)\s*:|\Z)",
            normalized_problem,
            flags=re.IGNORECASE | re.MULTILINE | re.DOTALL,
        )
        esw_turnaround = m_esw.group(1).strip() if m_esw else ""
    issue = _extract_subsection(normalized_problem, "Issue Part Number(s)", PROBLEM_SUBSECTION_NAMES)
    if not issue:
        issue = _extract_subsection(normalized_problem, "Issue Part Numbers", PROBLEM_SUBSECTION_NAMES)
    impact = _extract_subsection(normalized_problem, "Impact Details", PROBLEM_SUBSECTION_NAMES)

    rebuilt_problem_blocks: List[str] = []
    if ref:
        rebuilt_problem_blocks.append(change_drivers_header + "\n" + ref.strip())
    if esw_turnaround:
        rebuilt_problem_blocks.append(esw_turnaround_header + "\n" + _normalize_bullets(esw_turnaround, crisp=True))

    if not kept and original_lines:
        # Keep one line to avoid an empty Problem Description section.
        kept = [original_lines[0]]
        moved = moved[1:] if moved else moved
    kept_desc = _normalize_bullets("\n".join(kept)) if kept else "- Issue details are documented in source records."
    rebuilt_problem_blocks.append("Problem Description:\n" + kept_desc.strip())
    if impact:
        rebuilt_problem_blocks.append("Impact Details:\n" + _normalize_bullets(impact, crisp=True))
    rebuilt_problem_blocks.append("Issue Part Number(s):\n" + ((issue or "-").strip() or "-"))
    rebuilt_problem = "\n\n".join(rebuilt_problem_blocks).strip()

    # Merge moved bullets into Solution Description.
    existing_solution_desc = _extract_subsection(normalized_solution, "Solution Description", SOLUTION_SUBSECTION_NAMES)
    benefits = _extract_subsection(normalized_solution, "Benefits of the Proposed Solution", SOLUTION_SUBSECTION_NAMES)
    merged_solution = "\n".join([x.strip() for x in [existing_solution_desc, "\n".join(moved)] if x and x.strip()]).strip()
    merged_solution = _normalize_bullets(merged_solution)
    merged_solution = _split_inline_dashes_to_bullets(merged_solution)
    merged_solution = _rewrite_bullets_to_imperative(merged_solution, source_text)
    # Re-apply split in case the LLM collapsed bullets back to one line.
    merged_solution = _split_inline_dashes_to_bullets(merged_solution)
    # Strip any part descriptions the LLM may have re-introduced after rewriting.
    merged_solution = _remove_known_part_descriptions(merged_solution, source_text, remove_standalone=False)
    # LLM reframe: shorten any top-level bullet that still becomes too long.
    merged_solution = _reframe_long_solution_bullets(merged_solution, max_words=22)
    merged_solution = _complete_dangling_bullets(merged_solution, source_text, max_words=22, imperative=True)

    rebuilt_solution_blocks: List[str] = []
    if merged_solution:
        rebuilt_solution_blocks.append("Solution Description:\n" + merged_solution)
    if benefits:
        rebuilt_solution_blocks.append("Benefits of the Proposed Solution:\n" + _normalize_bullets(benefits, crisp=True))
    rebuilt_solution = "\n\n".join(rebuilt_solution_blocks).strip()

    return rebuilt_problem, rebuilt_solution


def _enforce_obsolescence_solution_style(
    solution_statement: str,
    problem_statement: str,
    source_text: str,
) -> str:
    """Enforce user-required obsolescence Solution Description bullet patterns.

        - If replacement is identified:
            1) Mention supplier-obsoleted MPN context and proposed replacement for impacted Applied P/N.
            2) State Applied Materials review/approval or alternate-identification responsibility.
            3) Check parent interchangeability; revise parent if interchangeable, or obsolete if FFF impacted.
        - If replacement is not identified:
            1) State supplier MPN obsolescence and Applied Materials alternate-identification responsibility.
            2) Revise/obsolete parents per interchangeability policy and FFF impact.
            3) Communicate approved/identified replacement planning with supplier.
    """
    if not solution_statement or not solution_statement.strip():
        return solution_statement

    lower = (source_text or "").lower()
    is_obs = any(k in lower for k in [
        "obsolete", "obsoleted", "obs part", "end of life", " eol", "eol ", "discontinue", "discontinued",
    ])
    if not is_obs:
        return solution_statement

    issue_pns = _extract_issue_part_numbers_from_problem(problem_statement)
    issue_pn = issue_pns[0] if issue_pns else ""

    # Detect replacement mappings from both the normalized source text and
    # the current solution statement so explicit pairs are never missed.
    pairs = extract_replacement_pairs(source_text)
    stmt_pairs = extract_replacement_pairs(solution_statement)
    all_pairs: List[Tuple[str, str]] = []
    for old_pn, new_pn in pairs + stmt_pairs:
        if not old_pn or not new_pn:
            continue
        pair = (old_pn.upper(), new_pn.upper())
        if pair not in all_pairs:
            all_pairs.append(pair)

    old_applied_pn = ""
    replacement_pn = ""
    if all_pairs:
        if issue_pn:
            for old_pn, new_pn in all_pairs:
                if old_pn.upper() == issue_pn.upper():
                    old_applied_pn = old_pn
                    replacement_pn = new_pn
                    break
        if not replacement_pn:
            old_applied_pn, replacement_pn = all_pairs[0]
    if not old_applied_pn:
        old_applied_pn = issue_pn

    # -------------------------------------------------------------------
    # Fallback: if extract_replacement_pairs found nothing, scan the full
    # solution statement (including Benefits the LLM already wrote) for
    # any AM part number that is explicitly labeled as a "Replacement" and
    # differs from the issue PN.  The LLM often correctly identifies the
    # replacement in Benefits even when the source text format does not
    # match any deterministic pair-extraction pattern.
    # -------------------------------------------------------------------
    if not replacement_pn:
        _all_solution_pns = extract_part_numbers(solution_statement)
        _issue_pn_set = {p.upper() for p in issue_pns}
        # Also collect all PNs already present in the problem statement so
        # we do not accidentally pick an affected-parent as replacement.
        _problem_pns_set = {p.upper() for p in extract_part_numbers(problem_statement)}

        # Strategy 1: Look for "Replacement P/N <PN>" or "Replacement <PN>"
        _repl_m = re.findall(
            rf"replacement\s+(?:P/?N\s+)?({PN_FRAGMENT})",
            solution_statement,
            flags=re.IGNORECASE,
        )
        for _candidate in _repl_m:
            _cu = _candidate.upper()
            if _cu not in _issue_pn_set and _cu not in _problem_pns_set:
                replacement_pn = _cu
                break

        # Strategy 2: Pick any AM PN in solution that is not in problem/issue
        if not replacement_pn:
            for _sp in _all_solution_pns:
                _su = _sp.upper()
                if _su not in _issue_pn_set and _su not in _problem_pns_set:
                    replacement_pn = _su
                    break

        if replacement_pn:
            old_applied_pn = issue_pn or old_applied_pn

    # Explicit replacement pair mapping always wins over textual "no replacement"
    # phrases that may exist in historical/source narrative.
    has_replacement = bool(replacement_pn)

    benefits = _extract_subsection(solution_statement, "Benefits of the Proposed Solution", SOLUTION_SUBSECTION_NAMES)

    if has_replacement:
        lead_issue = old_applied_pn or issue_pn or "the previous Applied P/N"
        bullets = [
            f"- Applied Materials has Qualified replacement {replacement_pn} as an alternate to Previous {lead_issue}.",
            f"- Obsolete {lead_issue} and replace it with Qualified replacement {replacement_pn}.",
            f"- Apply interchangeability policy to impacted parent structure for obsolete Applied P/N {lead_issue}.",
        ]
    else:
        lead_issue = issue_pn or (all_pairs[0][0] if all_pairs else "the OBS part")
        bullets = [
            f"- Replacement for Applied P/N {lead_issue} not found in input files.",
            "- Applied Materials must identify and qualify an alternate part meeting form-fit-function (FFF) requirements.",
            "- Perform required engineering validation and testing (fitment, performance, reliability).",
            "- Update all relevant documentation:",
            "  - BOM",
            "  - Drawings",
            "  - Specifications",
            "  - Work instructions",
            f"- Apply interchangeability policy to impacted parent structure for obsolete Applied P/N {lead_issue}.",
        ]

    blocks: List[str] = ["Solution Description:\n" + "\n".join(bullets)]
    if benefits:
        blocks.append("Benefits of the Proposed Solution:\n" + _normalize_bullets(benefits, crisp=True))
    return "\n\n".join(blocks).strip()

def _sc_shortage_deterministic_fallback(part_numbers: List[str], lead_pn: str) -> Dict[str, str]:
    """Returns a fully deterministic PSS when the LLM call fails or returns an error."""
    pn_str = ", ".join(part_numbers) if part_numbers else "the affected part"
    pn_ref = f"for part number {pn_str}" if part_numbers else "for the affected part"
    title = f"{lead_pn + ' - ' if lead_pn else ''}Supply Chain Criticality"
    problem = (
        "Problem Description:\n"
        f"- Critical Inventory Status: The supplier currently has no on-hand inventory {pn_ref},"
        " resulting in an immediate supply risk with no buffer available.\n"
        f"- Uncovered Future Demand: Demand requirements in upcoming quarters remain unfulfilled {pn_ref},"
        " creating a high likelihood of material shortages if no corrective action is taken.\n"
        f"- Supply Continuity Risk: With inventory fully dependent on scheduled receipts {pn_ref},"
        " any delay or disruption poses a direct threat to meeting future demand.\n"
        f"- Escalation-Level Item: The current supply condition for {pn_str} warrants elevated"
        " monitoring and cross-functional visibility due to potential downstream impact.\n\n"
        "Impact Details:\n"
        "- Supply constraints may affect production plans and customer commitments.\n"
        "- No buffer exists to absorb forecast changes or unexpected demand increases.\n"
        "- Downstream delivery risk increases without immediate corrective action.\n\n"
        "Issue Part Number(s):\n"
        f"{pn_str}"
    )
    solution = (
        "Solution Description:\n"
        f"- Initiate cross-functional supply chain escalation for {pn_str}.\n"
        "- Engage supplier to accelerate inbound supply and provide updated delivery commitment.\n"

        f"- Treat {pn_str} as a critical item until sustainable inventory levels and"

        "- Reduces supply continuity risk through proactive escalation and supplier engagement.\n"
        "- Establishes a recovery path to restore inventory coverage and production stability.\n"
        "- Improves cross-functional visibility and responsiveness to supply constraints."
    )
    return {
        "title": title[:85],
        "problem_statement": problem,
        "solution_statement": solution,
        "raw": "",
    }


def _generate_sc_shortage_pss(part_numbers: List[str], source_text: str, payload: Dict[str, Any]) -> Dict[str, str]:
    """Dedicated PSS generator for supply chain shortage / inventory-demand mismatch scenarios.

    ARCHITECTURE: The LLM receives ONLY part numbers — never the raw source text.
    This completely prevents reproduction of inventory counts, demand figures, PO numbers,
    dates, or quarter-coverage details from the input.
    """
    lead_pn = part_numbers[0] if part_numbers else ""
    pn_list = ", ".join(part_numbers) if part_numbers else "the affected part"
    title_prefix = f"{lead_pn} - " if lead_pn else ""

    # Detect scenario type from source_text to shape contextually appropriate output.
    # Note: source_text has already had PO numbers stripped by generate_full_pss.
    lower_src = source_text.lower()
    is_eol = any(kw in lower_src for kw in [
        "end of life", " eol", "eol ", "eol)", "(eol", "end-of-life",
        "end of production", "discontinue", "discontinued",
    ])
    no_replacement_yet = any(kw in lower_src for kw in [
        "no replacement", "not yet identified", "no approved replacement",
        "replacement p/n not", "replacement not identified", "replacement not communicated",
        "no drop-in", "no drop in", "not identified or communicated",
    ])
    has_replacement = (not no_replacement_yet) and any(kw in lower_src for kw in [
        "replacement", "alternate part", "alternate p/n", "alternate pn",
    ])

    # Build scenario context block: non-numerical facts the LLM can use to frame output.
    scenario_context_lines: List[str] = []
    if is_eol:
        scenario_context_lines.append(
            f"- Part number {lead_pn or pn_list} has been declared End of Life (EOL) by the OEM manufacturer."
        )
        if no_replacement_yet:
            scenario_context_lines.append(
                "- No approved drop-in replacement part number has been identified or communicated as of this ECR."
            )
        elif has_replacement:
            scenario_context_lines.append(
                "- A replacement part has been identified and requires qualification and BOM update."
            )
    scenario_context = (
        "SCENARIO CONTEXT (non-numerical facts — use to frame the output):\n"
        + "\n".join(scenario_context_lines)
        if scenario_context_lines else ""
    )

    # EOL-specific solution actions (only injected when EOL is detected).
    eol_solution_actions = ""
    if is_eol:
        eol_solution_actions = """\nADDITIONAL EOL SOLUTION ACTIONS — include 2 to 3 of these:
  - Identify and qualify an approved drop-in replacement part number for the EOL component.
  - Communicate the approved replacement part number to the supplier to enable procurement planning.
  - Update the Bill of Materials (BOM) for all affected parent assemblies to reflect the approved replacement.
  - Disposition all outstanding procurement obligations for the EOL component in alignment with the approved replacement.
  - Engage the OEM or alternate approved suppliers to confirm replacement component availability and lead time."""

    # Default title differs by scenario type.
    default_title_body = "OEM EOL Declaration — Supply Continuity Risk" if is_eol else "Supply Chain Criticality — Inventory & Demand Risk"
    default_title = f"{title_prefix}{default_title_body}"

    # Pre-select the single best-matching SC statement from the source text.
    _sc_label, _sc_selected_stmt = _select_sc_generic_statement(source_text)

    prompt = f"""You are a Supply Chain Risk Communication assistant writing a formal Engineering Change Request (ECR) Problem and Solution Statement.

ABSOLUTE RULES — ZERO EXCEPTIONS — VIOLATIONS WILL INVALIDATE THE ENTIRE OUTPUT:
1. DO NOT mention ANY inventory quantities, counts, or numeric supply values.
2. DO NOT mention ANY demand quantities or numeric demand figures.
3. DO NOT mention ANY PO numbers, Purchase Order numbers, or order reference numbers.
4. DO NOT mention ANY specific dates, quarters (e.g., 25Q3, Q4 2025), or time periods.
5. DO NOT reproduce ANY numbers from the input — not even masked or paraphrased.
6. Use ONLY the approved statement and solution actions listed below.
7. ALWAYS reference the part number(s) listed below in your output.
8. Do NOT invent facts, supplier commitments, or new information beyond what is stated in Scenario Context.

ISSUE PART NUMBER(S): {pn_list}
{(chr(10) + scenario_context) if scenario_context else ""}

APPROVED SUPPLY CHAIN CRITICALITY STATEMENT — USE VERBATIM IN PROBLEM DESCRIPTION:
  - {_sc_selected_stmt}
    Pair with the issue part number(s). Use this statement exactly as written.

APPROVED SOLUTION ACTIONS — USE 2 TO 4 IN SOLUTION DESCRIPTION:
    - Initiate cross-functional supply chain escalation for the issue part number(s).
  - Engage supplier to accelerate inbound supply and provide an updated delivery commitment.
    - Evaluate alternative sourcing options or qualified alternate suppliers for the issue part number(s).
  - Implement demand smoothing measures to prioritize coverage for critical production plans.
  - Establish enhanced monitoring cadence until sustainable inventory levels are restored.
    - Treat the issue part number(s) as critical items until forward coverage is re-established.{eol_solution_actions}

OUTPUT FORMAT — use EXACTLY these headers, no deviations:
Title:
{default_title}

    "Problem Statement:\n"
    "Problem Description:\n"
    "<3 to 5 bullet points — use ONLY approved generic statements above — always reference the part number(s) — NO quantities, NO PO numbers, NO dates>\n"

    "Impact Details:\n"
    "<2 to 3 short bullets on operational/business risk — NO numbers>\n\n"

    "Issue Part Number(s):\n"
    f"{pn_list}\n"

Solution Statement:
Solution Description:
<2 to 4 bullet points — use ONLY approved solution actions above — NO quantities, NO PO numbers>

Benefits of the Proposed Solution:
<2 to 3 short bullets on expected improvement — NO numbers>

REMINDER: Your output must contain ZERO inventory numbers, ZERO demand figures, ZERO PO numbers, ZERO quarter or date references.
Do NOT include any numeric sequence that resembles an order number, reference number, or identifier from the input.
Use ONLY the exact approved statements listed above. Do NOT add new sentences that introduce numbers or order references.""".strip()

    drafted = _call_databricks(prompt)
    if not drafted or drafted.startswith("Error:"):
        return _sc_shortage_deterministic_fallback(part_numbers, lead_pn)

    drafted = clean_ai_output(drafted)

    # Extract structured sections
    raw_title = _extract_title(drafted).strip()
    title = raw_title if raw_title else default_title
    # Ensure the lead part number prefix is present in the title
    if lead_pn and not title.startswith(lead_pn):
        body = re.sub(rf"^{re.escape(lead_pn)}\s*[-\u2013]?\s*", "", title).strip(" -\u2013")
        title = f"{lead_pn} - {body}" if body else default_title
    title = title[:85]

    problem_section = _extract_section(drafted, "Problem Statement").strip()
    solution_section = (
        _extract_section(drafted, "Solution Statement").strip()
        or _extract_section(drafted, "Proposed Solution").strip()
    )

    # Hard post-processing: strip any numbers that leaked through
    problem_section = _sanitize_supply_chain_numbers(problem_section)
    solution_section = _sanitize_supply_chain_numbers(solution_section)

    # Ensure part numbers are present in the output
    if part_numbers and not any(pn in problem_section for pn in part_numbers):
        problem_section = f"{problem_section}\n\nIssue Part Number(s):\n{pn_list}"

    return {
        "title": title,
        "problem_statement": problem_section,
        "solution_statement": solution_section,
        "raw": drafted[:12000],
    }



def _regroup_problem_description(problem_text: str, source_text: str = "") -> str:
    """
    PLM Final Step: Groups similar items in Problem Description by change category.

    Reads the already-written Problem Description, checks for grouping possibilities
    (multiple supplier-driven changes spanning different component/firmware/hardware categories),
    and rewrites the content grouped under numbered category headers.

    Runs LAST in the processing pipeline, after all other normalizations and LLM passes.
    Only triggers when 4 or more top-level bullet items are present in Problem Description.
    All other subsections (Change Drivers, ESW Turn Around, Impact Details, Issue Part Number(s))
    are preserved unchanged.
    """
    if not problem_text or not problem_text.strip():
        return problem_text

    # Extract Problem Description subsection only
    problem_desc = _extract_subsection(problem_text, "Problem Description", PROBLEM_SUBSECTION_NAMES)
    if not problem_desc or not problem_desc.strip():
        return problem_text

    # Count top-level bullets — only regroup when there are enough distinct items
    top_bullet_lines = [
        ln for ln in problem_desc.splitlines()
        if re.match(r"^\s*-\s+\S", ln) and not re.match(r"^\s{3,}-\s+", ln)
    ]
    if len(top_bullet_lines) < 4:
        return problem_text  # Not enough content to meaningfully group

    prompt = (
        "You are a senior manufacturing engineer finalizing an ECR Problem Description for PLM submission.\n\n"
        "TASK: Review the Problem Description below and rewrite it by GROUPING similar change items "
        "under numbered category headers. This is the final formatting step before PLM entry.\n\n"
        "STRICT RULES:\n"
        "1. Write 1-2 introductory sentence(s) as plain prose (no bullet) establishing: "
        "part name, Applied Materials P/N (if present), supplier name, and overall change reason.\n"
        "2. Group all change details under numbered category headers. "
        "Use ONLY categories that have actual content in the input:\n"
        "   1. Supplier-Driven Obsolescence\n"
        "   2. Electronic Component Changes\n"
        "   3. Mechanical / Hardware Changes\n"
        "   4. Firmware / Software Changes\n"
        "   5. Process / Design Changes\n"
        "3. Under each category header, list each change as a sub-bullet starting with '- '.\n"
        "4. For component changes use format:\n"
        "   - Old Manufacturer (MPN old) changed to New Manufacturer (MPN new): description\n"
        "   - OR: MPN old changed to MPN new: description (if same manufacturer)\n"
        "   - Use 'From: <old value> To: <new value>' for version/revision/firmware changes.\n"
        "5. Applied Materials P/N (NNNN-NNNNN format) must appear ONLY in the introductory "
        "summary. Do NOT repeat it in any category sub-bullet — reference 'the part', "
        "'this component', or the supplier MPN instead.\n"
        "6. PLAIN TEXT ONLY: No arrows (->), Unicode special characters, or markdown formatting. "
        "Use only plain hyphens (-) for sub-bullets.\n"
        "7. Preserve ALL factual data exactly: part numbers, manufacturer names, version numbers, "
        "MPN codes, descriptions.\n"
        "8. Do NOT add assumptions, generic statements, or content not in the input.\n"
        "9. Do NOT include a numbered category if it has no items.\n"
        "10. If only one category applies (all changes are the same type), write that one "
        "category with its sub-bullets.\n\n"
        "OUTPUT FORMAT — use exactly this structure:\n"
        "<Introductory summary sentence(s) as plain prose>\n\n"
        "N. <Category Name>\n"
        "- <change sub-bullet>\n"
        "- <change sub-bullet>\n\n"
        "(repeat for each applicable numbered category)\n\n"
        "PROBLEM DESCRIPTION TO REGROUP:\n"
        f"{problem_desc.strip()}"
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return problem_text

    # clean_ai_output strips markdown bold/italic, heading markers, etc.
    regrouped = clean_ai_output(result.strip())
    if not regrouped:
        return problem_text

    # Preserve all other subsections; only replace Problem Description content.
    m_ref_header = re.search(
        r"^\s*(Change\s+Drivers(?:\s*\([^\n)]*\))?)\s*:",
        problem_text,
        flags=re.IGNORECASE | re.MULTILINE,
    )
    change_drivers_header = m_ref_header.group(1).strip() + ":" if m_ref_header else "Change Drivers:"

    m_esw_header = re.search(
        r"^\s*(ESW\s+Turn\s+Around(?:\s*\([^\n)]*\))?)\s*:",
        problem_text,
        flags=re.IGNORECASE | re.MULTILINE,
    )
    esw_turnaround_header = m_esw_header.group(1).strip() + ":" if m_esw_header else "ESW Turn Around:"

    ref = _extract_subsection(problem_text, "Change Drivers", PROBLEM_SUBSECTION_NAMES)
    esw_turnaround = _extract_subsection(problem_text, "ESW Turn Around", PROBLEM_SUBSECTION_NAMES)
    impact_details = _extract_subsection(problem_text, "Impact Details", PROBLEM_SUBSECTION_NAMES)
    issue_parts = _extract_subsection(problem_text, "Issue Part Number(s)", PROBLEM_SUBSECTION_NAMES)
    if not issue_parts:
        issue_parts = _extract_subsection(problem_text, "Issue Part Numbers", PROBLEM_SUBSECTION_NAMES)

    blocks: List[str] = []
    if ref:
        blocks.append(change_drivers_header + "\n" + ref.strip())
    if esw_turnaround:
        blocks.append(esw_turnaround_header + "\n" + esw_turnaround.strip())
    blocks.append("Problem Description:\n" + regrouped.strip())
    if impact_details:
        blocks.append("Impact Details:\n" + impact_details.strip())
    blocks.append("Issue Part Number(s):\n" + (issue_parts.strip() if issue_parts else "-"))

    return "\n\n".join(blocks).strip()


def generate_full_pss(payload: Dict[str, Any]) -> Dict[str, str]:
    source_text = _compose_user_text(payload)
    # Apply source-based revision ordering early, before other normalization
    source_text = _apply_source_based_from_to_order(source_text, payload)
    if not source_text:
        return {"title": "", "problem_statement": "", "solution_statement": "", "raw": ""}

    # ALWAYS strip PO numbers before any LLM call — applies to every scenario, not just SC.
    # This ensures PO numbers from the input can never be reproduced in any output.
    source_text = _SC_PO_PATTERN.sub("", source_text)
    # Clean up only horizontal whitespace (spaces/tabs) left by the PO strip.
    # Do NOT touch newlines — they preserve the section structure built by _compose_user_text.
    source_text = re.sub(r",[ \t]*,", ",", source_text)
    source_text = re.sub(r"[ \t]{2,}", " ", source_text).strip()

    is_sc = _is_supply_chain_shortage_scenario(source_text)
    if is_sc:
        # Bypass the normal engineering prompt entirely.
        # Extract part numbers from source_text (PO numbers already stripped above).
        # The SC generator sends ONLY part numbers and scenario context to the LLM —
        # no inventory figures, demand quantities, or PO numbers ever reach the LLM.
        sc_part_numbers = extract_part_numbers(source_text)
        return _generate_sc_shortage_pss(sc_part_numbers, source_text, payload)

    drafted = reframe_problem(source_text)
    drafted = str(drafted or "").strip()

    if not drafted or drafted.startswith("Error:"):
        return {
            "title": str(payload.get("short_title") or "").strip()[:200],
            "problem_statement": (payload.get("current_problem_text") or source_text),
            "solution_statement": str(payload.get("current_solution_text") or payload.get("proposed_solution_text") or ""),
            "raw": drafted or "",
        }

    title = _normalize_title_with_part_number(_extract_title(drafted).strip(), source_text)
    problem_section = _extract_section(drafted, "Problem Statement").strip()
    solution_section = _extract_section(drafted, "Solution Statement").strip()
    if not solution_section:
        solution_section = _extract_section(drafted, "Proposed Solution").strip()

    normalized_problem = _normalize_problem_statement(problem_section, source_text)
    normalized_solution = _normalize_solution_statement(solution_section, source_text)

    payload_reference_summary = _build_reference_change_summary_from_payload(payload)

    # Build Change Drivers header with PCR/Project numbers included
    pcr_ids = [str(rec.get("pcr_id", "")) for rec in payload.get("pcr_records") or [] if rec.get("pcr_id")]
    project_ids = [str(rec.get("project_id", "")) for rec in payload.get("project_records") or [] if rec.get("project_id")]
    id_parts = [f"PCR# {pid}" for pid in pcr_ids] + [f"Project# {pid}" for pid in project_ids]
    cd_header = "Change Drivers (" + ", ".join(id_parts) + "):" if id_parts else "Change Drivers:"

    # ESW Turn Around goes immediately below Change Drivers, before rest of problem
    esw_turnaround_block = _build_esw_turnaround_from_payload(payload)

    # Always show Change Drivers header if PCR/Project records exist
    show_cd = bool(pcr_ids or project_ids)
    # Always show ESW Turn Around header if ESW records exist
    show_esw = bool(payload.get("esw_records"))

    cd_block = cd_header
    if payload_reference_summary:
        cd_block += "\n" + payload_reference_summary
    elif show_cd:
        cd_block += "\n(No summary available)"

    esw_block = esw_turnaround_block if esw_turnaround_block else ("ESW Turn Around:" if show_esw else "")

    top_block = ""
    if show_cd:
        top_block = cd_block
    if show_esw:
        top_block = (top_block + "\n\n" + esw_block).strip() if top_block else esw_block

    if top_block:
        normalized_problem = (
            top_block
            + ("\n\n" + normalized_problem if normalized_problem else "")
        ).strip()

    # Always rebalance — LLM can mix problem/solution content regardless of source fields.
    normalized_problem, normalized_solution = _rebalance_problem_solution(
        normalized_problem,
        normalized_solution,
        source_text,
    )

    # Keep reframed LLM output as primary; use deterministic source content only as fallback.
    if not normalized_problem:
        normalized_problem = _build_complete_problem_description(payload)

    # NOTE: Raw source-line backfill into Problem Description is intentionally disabled.
    # It can inject long copy-pasted source sentences that violate the expected format.

    normalized_solution = _build_complete_solution_statement(payload, normalized_solution)

    # Enforce user rule: Title must reference Issue P/N (single issue only).
    issue_pns_for_title = _extract_issue_part_numbers_from_problem(normalized_problem)
    if len(issue_pns_for_title) == 1:
        title = _normalize_title_with_part_number(
            title,
            source_text,
            lead_pn_override=issue_pns_for_title[0],
            force_no_prefix=False,
        )
    elif len(issue_pns_for_title) > 1:
        title = _normalize_title_with_part_number(
            title,
            source_text,
            lead_pn_override="",
            force_no_prefix=True,
        )

    # Enforce obsolescence-specific Solution Description bullet pattern.
    normalized_solution = _enforce_obsolescence_solution_style(
        normalized_solution,
        normalized_problem,
        source_text,
    )

    title = _apply_engineering_shortcuts(title)
    normalized_problem = _apply_engineering_shortcuts(normalized_problem)
    normalized_solution = _apply_engineering_shortcuts(normalized_solution)

    # Apply source-based From/To ordering correction before removing revision tokens
    normalized_problem = _apply_source_based_from_to_order(normalized_problem, payload)
    normalized_solution = _apply_source_based_from_to_order(normalized_solution, payload)
    
    # Remove supplier revision notation from Problem and Solution descriptions per requirements.
    normalized_problem = _remove_supplier_rev_tokens(normalized_problem)
    normalized_solution = _remove_supplier_rev_tokens(normalized_solution)

    # Formatting guardrail: maintain exactly one blank line between sections.
    normalized_problem = _enforce_single_blank_line_between_sections(normalized_problem)
    normalized_solution = _enforce_single_blank_line_between_sections(normalized_solution)

    # PLM Final Step 1: Group similar change items in Problem Description under numbered
    # category headers (Electronic, Mechanical, Firmware, etc.). Runs last so it operates
    # on the fully-normalized output. Only activates when 4+ top-level bullets exist.
    normalized_problem = _regroup_problem_description(normalized_problem, source_text)

    # PLM Final Step 2: Apply bold formatting to From: and To: labels throughout both
    # statements. Deterministic post-processor — runs after all LLM calls and
    # clean_ai_output invocations so bold markers are preserved in the final output.
    normalized_problem = _bold_from_to_values(normalized_problem)
    normalized_solution = _bold_from_to_values(normalized_solution)

    # Fallback: if parsing extracted nothing despite the LLM returning content,
    # use the raw drafted text so the caller never gets all-empty fields.
    if not any([title, normalized_problem, normalized_solution]) and drafted:
        return {
            "title": str(payload.get("short_title") or "").strip()[:200],
            "problem_statement": drafted,
            "solution_statement": str(payload.get("current_solution_text") or payload.get("proposed_solution_text") or ""),
            "raw": drafted[:12000],
        }

    return {
        "title": title[:85],
        "problem_statement": normalized_problem,
        "solution_statement": normalized_solution,
        "raw": drafted[:12000],
    }


def generate_problem_summary(payload: Dict[str, Any]) -> str:

def generate_pss(payload: Dict[str, Any]) -> str:

def classify_reason_code_with_llm(problem: str, solution: str, return_evidence: bool = False) -> tuple:
    """Use the LLM to propose the correct ECO Reason Code.

    Builds a structured prompt embedding the full ECM classification rules,
    sends it to the Databricks LLM endpoint, and parses the strict response.

    Returns:
        - Default: (code_name, justification)
        - If return_evidence=True: (code_name, justification, evidence_list)

        code_name is the display name (e.g. 'Design Correction') and
        justification is the LLM rationale. evidence_list contains up to
        three short supporting snippets copied from input text.

        Returns empty values on any error so the caller can fall back gracefully.
    """
    if not problem.strip() and not solution.strip():
        return ("", "", []) if return_evidence else ("", "")

    prompt = _RC_CLASSIFICATION_SYSTEM_PROMPT.replace(
        "{problem_description}", problem.strip()
    ).replace(
        "{solution_description}", solution.strip()
    )

    result = _call_databricks(prompt)
    if not result or result.startswith("Error:"):
        return ("", "", []) if return_evidence else ("", "")

    rc_num = ""
    justification = ""
    evidence: List[str] = []
    for line in result.strip().splitlines():
        line = line.strip()
        if line.lower().startswith("reason code:"):
            raw_rc = line.split(":", 1)[1].strip()
            # Accept "4.4", "4.4 – Design Correction", "4.4 Design Correction", etc.
            import re as _re
            m = _re.match(r"(4\.\d{1,2})", raw_rc)
            if m:
                rc_num = m.group(1)
        elif line.lower().startswith("justification:"):
            justification = line.split(":", 1)[1].strip()
        elif line.lower().startswith("evidence:"):
            raw_evidence = line.split(":", 1)[1].strip()
            if raw_evidence:
                pieces = [p.strip() for p in raw_evidence.split("||") if p.strip()]
                cleaned: List[str] = []
                for p in pieces:
                    p = p.strip().strip('"').strip("'").strip()
                    # Remove leading bullet-style prefixes if present.
                    p = re.sub(r"^[-*•\s]+", "", p)
                    if p:
                        cleaned.append(p)
                evidence = cleaned[:3]

    if not rc_num:
        return ("", "", []) if return_evidence else ("", "")

    code_name = _RC_NUM_TO_NAME.get(rc_num, rc_num)
    if return_evidence:
        return code_name, justification, evidence
    return code_name, justification

def read_email_file(file_path: str) -> str:
    """
    Read reference attachment content and extract readable text.
    - .msg  : parsed via extract_msg (Outlook binary format)
    - .eml  : parsed via Python's email stdlib
    - .txt  : read as plain text
    - .pdf  : parsed via pypdf / PyPDF2
    - .pptx : parsed via python-pptx
    - .docx : parsed via python-docx
    - .ppt / .doc : best-effort plain-text fallback
    """
    if not file_path:
        return ""

    ext = os.path.splitext(file_path)[1].lower()

    # --- Outlook .msg ---
    if ext == ".msg":
        try:
            import extract_msg
            with extract_msg.openMsg(file_path) as msg:
                parts = []
                if msg.subject:
                    parts.append(f"Subject: {msg.subject.strip()}")
                if msg.sender:
                    parts.append(f"From: {msg.sender.strip()}")
                if msg.date:
                    parts.append(f"Date: {msg.date}")
                body = (msg.body or "").strip()
                if not body:
                    # fallback to HTML body stripped of tags
                    html = (msg.htmlBody or b"").decode("utf-8", errors="ignore")
                    body = re.sub(r"<[^>]+>", " ", html)
                    body = re.sub(r"[ \t]+", " ", body).strip()
                if body:
                    parts.append(f"\n{body}")
                return "\n".join(parts).strip()
        except Exception as e:
            return f"Error reading .msg file: {e}"

    # --- Standard .eml ---
    if ext == ".eml":
        try:
            import email as _email
            import email.policy
            with open(file_path, "rb") as f:
                msg = _email.message_from_binary_file(f, policy=_email.policy.default)
            parts = []
            for hdr in ("Subject", "From", "Date"):
                val = msg.get(hdr, "").strip()
                if val:
                    parts.append(f"{hdr}: {val}")
            # Walk MIME parts for plain text
            body_parts = []
            for part in msg.walk():
                if part.get_content_type() == "text/plain":
                    payload = part.get_payload(decode=True)
                    if payload:
                        charset = part.get_content_charset() or "utf-8"
                        body_parts.append(payload.decode(charset, errors="ignore"))
            if body_parts:
                parts.append("\n" + "\n".join(body_parts).strip())
            return "\n".join(parts).strip()
        except Exception as e:
            return f"Error reading .eml file: {e}"

    # --- PDF ---
    if ext == ".pdf":
        try:
            text_chunks = []
            try:
                from pypdf import PdfReader  # type: ignore
            except Exception:
                from PyPDF2 import PdfReader  # type: ignore

            reader = PdfReader(file_path)
            for page in reader.pages:
                txt = page.extract_text() or ""
                if txt.strip():
                    text_chunks.append(txt)
            return "\n\n".join(text_chunks).strip()
        except Exception as e:
            return f"Error reading .pdf file: {e}"

    # --- PowerPoint ---
    if ext == ".pptx":
        try:
            from pptx import Presentation  # type: ignore

            prs = Presentation(file_path)
            slide_text = []
            for idx, slide in enumerate(prs.slides, start=1):
                lines = []
                for shape in slide.shapes:
                    if hasattr(shape, "text") and shape.text:
                        t = str(shape.text).strip()
                        if t:
                            lines.append(t)
                if lines:
                    slide_text.append(f"Slide {idx}:\n" + "\n".join(lines))
            return "\n\n".join(slide_text).strip()
        except Exception as e:
            return f"Error reading .pptx file: {e}"

    # --- Word ---
    if ext == ".docx":
        try:
            from docx import Document  # type: ignore

            doc = Document(file_path)
            lines = [p.text.strip() for p in doc.paragraphs if p.text and p.text.strip()]
            return "\n".join(lines).strip()
        except Exception as e:
            return f"Error reading .docx file: {e}"

    # --- Legacy office fallback ---
    if ext in {".doc", ".ppt"}:
        try:
            with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
                txt = f.read().strip()
            if txt:
                return txt
            return (
                f"Error reading {ext} file: format is not directly supported. "
                f"Please save as .docx/.pptx and retry."
            )
        except Exception as e:
            return f"Error reading {ext} file: {e}"

    # --- Plain text / fallback ---
    try:
        with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
            return f.read().strip()
    except Exception as e:
        return f"Error reading email file: {e}"


def extract_problem_from_email(email_content: str) -> str:
    """
    Parse email content and extract the problem statement.
    Looks for issue/problem/complaint descriptions in the email body.
    """
    if not email_content:
        return ""
    
    lines = email_content.split('\n')
    problem_lines = []
    in_body = False
    
    for i, line in enumerate(lines):
        # Skip email headers
        if line.strip() == "":
            in_body = True
            continue
        
        if not in_body:
            continue
        
        # Stop at signature or forwarded message
        if re.search(r'^(--|_____|==|From:|Sent:|To:|Cc:)', line):
            break
        
        problem_lines.append(line)
    
    problem_text = '\n'.join(problem_lines).strip()
    # Take first 1000 chars of problem description
    return problem_text[:1000] if problem_text else ""


def extract_solution_from_email(email_content: str) -> str:
    """
    Parse email content and extract the proposed solution.
    Looks for solution/recommendation/action items that are approved/final.
    Ignores alternative or rejected solutions.
    """
    if not email_content:
        return ""
    
    # Find solution-related sections
    solution_patterns = [
        r'(?:solution|proposed|recommend|action|approved|final)[:\s]+(.+?)(?=(?:alternative|option|other|instead|but|however|note|ps:|--)|$)',
    ]
    
    solution_text = ""
    for pattern in solution_patterns:
        matches = re.findall(pattern, email_content, flags=re.IGNORECASE | re.DOTALL)
        if matches:
            solution_text = matches[0].strip()
            break
    
    # If no explicit solution found, look for approved recommendations
    if not solution_text:
        # Look for "approved" or "final" solutions
        approved_match = re.search(
            r'(?:approved|final|confirmed|decided)[\s:]+(.+?)(?=(?:alternative|option|rejected|dismissed|note|ps:|--)|$)',
            email_content,
            flags=re.IGNORECASE | re.DOTALL
        )
        if approved_match:
            solution_text = approved_match.group(1).strip()
    
    # Clean up multiline solution text
    solution_text = re.sub(r'\n{2,}', ' ', solution_text)
    solution_text = re.sub(r'\s+', ' ', solution_text).strip()
    
    return solution_text[:1000] if solution_text else ""


def correlate_email_with_problem(email_content: str, user_problem: str, payload: Dict[str, Any]) -> Dict[str, str]:
    """
    Read email, correlate with user problem/solution inputs, and generate PSS output.
    The full email text is always passed to the AI so it can extract context itself.
    """
    email_text = (email_content or "").strip()
    user_text = (user_problem or "").strip()

    # Guard: if email reading itself returned an error, surface it early
    if email_text.startswith("Error reading"):
        return {
            "title": "", "problem_statement": "", "solution_statement": "",
            "raw": email_text,
            "error": email_text,
        }

    # Get user's existing proposed solution from payload
    user_solution = str(
        payload.get("current_solution_text") or payload.get("proposed_solution_text") or ""
    ).strip()

    # Always include the full email so the AI can read it directly
    input_parts = []
    if email_text:
        input_parts.append(f"Email Content:\n{email_text[:3000]}")
    if user_text:
        input_parts.append(f"User Entered Problem:\n{user_text}")
    if user_solution:
        input_parts.append(f"User Proposed Solution:\n{user_solution}")

    combined_input = "\n\n".join(input_parts)
    
    # Build payload for AI processing
    email_payload = dict(payload)
    email_payload["current_problem_text"] = combined_input
    email_payload["email_content"] = email_text
    # Ensure user's proposed solution is preserved in payload
    if user_solution and not email_payload.get("proposed_solution_text"):
        email_payload["proposed_solution_text"] = user_solution
    
    # Generate PSS from correlated content
    return generate_full_pss(email_payload)
